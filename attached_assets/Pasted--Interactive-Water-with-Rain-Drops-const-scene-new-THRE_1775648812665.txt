  THREE.MathUtils.degToRad(-53),
  THREE.MathUtils.degToRad(-30),
  THREE.MathUtils.degToRad(-34)

// Water geometry
const SIZE = 80;
const SEGMENTS = 200;
const geometry = new THREE.PlaneGeometry(SIZE, SIZE, SEGMENTS, SEGMENTS);
geometry.rotateX(-Math.PI / 2);

const vertexCount = (SEGMENTS + 1) * (SEGMENTS + 1);
const currentHeight = new Float32Array(vertexCount);
const previousHeight = new Float32Array(vertexCount);
const velocity = new Float32Array(vertexCount);

// Water shader material
const waterMaterial = new THREE.ShaderMaterial({
  uniforms: {
    uTime: { value: 0 },
    uDeepColor: { value: new THREE.Color(0x1a3a5c) },
    uShallowColor: { value: new THREE.Color(0x4a7a9b) },
    uFoamColor: { value: new THREE.Color(0x8ab4cc) },
    uLightDir: { value: new THREE.Vector3(0.3, 0.7, 0.2).normalize() },
    uCameraPos: { value: camera.position },
    uFresnelPower: { value: 3.0 },
    uSpecularPower: { value: 128.0 },
  },
  vertexShader: `
    varying vec3 vWorldPosition;
    varying vec3 vNormal;
    varying float vHeight;
    
    void main() {
      vWorldPosition = (modelMatrix * vec4(position, 1.0)).xyz;
      vNormal = normalize(normalMatrix * normal);
      vHeight = position.y;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  fragmentShader: `
    uniform float uTime;
    uniform vec3 uDeepColor;
    uniform vec3 uShallowColor;
    uniform vec3 uFoamColor;
    uniform vec3 uLightDir;
    uniform vec3 uCameraPos;
    uniform float uFresnelPower;
    uniform float uSpecularPower;
    
    varying vec3 vWorldPosition;
    varying vec3 vNormal;
    varying float vHeight;
    
    void main() {
      vec3 normal = normalize(vNormal);
      vec3 viewDir = normalize(uCameraPos - vWorldPosition);
      
      // Fresnel
      float fresnel = pow(1.0 - max(dot(viewDir, normal), 0.0), uFresnelPower);
      
      // Diffuse
      float diffuse = max(dot(normal, uLightDir), 0.0) * 0.5 + 0.5;
      
      // Specular (Blinn-Phong)
      vec3 halfDir = normalize(uLightDir + viewDir);
      float spec = pow(max(dot(normal, halfDir), 0.0), uSpecularPower);
      
      // Height-based color
      float heightFactor = smoothstep(-0.5, 1.5, vHeight);
      vec3 waterColor = mix(uDeepColor, uShallowColor, heightFactor);
      
      // Foam on peaks
      float foam = smoothstep(0.4, 1.0, vHeight);
      waterColor = mix(waterColor, uFoamColor, foam * 0.5);
      
      // Combine
      vec3 skyColor = vec3(0.3, 0.4, 0.55);
      vec3 finalColor = mix(waterColor * diffuse, skyColor, fresnel * 0.45);
      finalColor += vec3(0.7, 0.8, 0.9) * spec * 1.8;
      finalColor += uFoamColor * foam * 0.12;
      
      gl_FragColor = vec4(finalColor, 0.95);
    }
  `,
  transparent: true,
  side: THREE.DoubleSide,
});

const waterMesh = new THREE.Mesh(geometry, waterMaterial);
scene.add(waterMesh);

// ============ RAIN SYSTEM ============
const RAIN_COUNT = 444;
const RAIN_AREA = 90;
const RAIN_HEIGHT_MIN = 0;
const RAIN_HEIGHT_MAX = 70;

const rainGeo = new THREE.BufferGeometry();
const rainPositions = new Float32Array(RAIN_COUNT * 3);
const rainVelocities = new Float32Array(RAIN_COUNT);
const rainOpacities = new Float32Array(RAIN_COUNT);

for (let i = 0; i < RAIN_COUNT; i++) {
  rainPositions[i * 3] = (Math.random() - 0.5) * RAIN_AREA;
  rainPositions[i * 3 + 1] = Math.random() * RAIN_HEIGHT_MAX + RAIN_HEIGHT_MIN;
  rainPositions[i * 3 + 2] = (Math.random() - 0.5) * RAIN_AREA;
  rainVelocities[i] = 30 + Math.random() * 30;
  rainOpacities[i] = 0.2 + Math.random() * 0.5;
}

rainGeo.setAttribute('position', new THREE.BufferAttribute(rainPositions, 3));
rainGeo.setAttribute('aOpacity', new THREE.BufferAttribute(rainOpacities, 1));

const rainMaterial = new THREE.ShaderMaterial({
  uniforms: {
    uColor: { value: new THREE.Color(0x8aaabe) },
  },
  vertexShader: `
    attribute float aOpacity;
    varying float vOpacity;
    void main() {
      vOpacity = aOpacity;
      vec4 mvPos = modelViewMatrix * vec4(position, 1.0);
      gl_Position = projectionMatrix * mvPos;
      gl_PointSize = max(1.0, 2.5 * (200.0 / -mvPos.z));
    }
  `,
  fragmentShader: `
    uniform vec3 uColor;
    varying float vOpacity;
    void main() {
      vec2 c = gl_PointCoord - 0.5;
      float d = length(c);
      if (d > 0.5) discard;
      gl_FragColor = vec4(uColor, vOpacity * (1.0 - d * 2.0));
    }
  `,
  transparent: true,
  depthWrite: false,
  blending: THREE.AdditiveBlending,
});

const rain = new THREE.Points(rainGeo, rainMaterial);
scene.add(rain);

// Rain streaks (lines for fast-falling rain effect)

  streakPositions[idx + 1] = y;
  streakPositions[idx + 2] = z;

  streakPositions[idx + 4] = y - len;
  streakPositions[idx + 5] = z;

streakGeo.setAttribute('position', new THREE.BufferAttribute(streakPositions, 3));

  color: 0x7a9ab0,

// ============ SPLASH PARTICLES ============
const MAX_SPLASHES = 600;
const splashGeo = new THREE.BufferGeometry();
const splashPositions = new Float32Array(MAX_SPLASHES * 3);
const splashSizes = new Float32Array(MAX_SPLASHES);
const splashAlphas = new Float32Array(MAX_SPLASHES);

splashGeo.setAttribute('position', new THREE.BufferAttribute(splashPositions, 3));
splashGeo.setAttribute('aSize', new THREE.BufferAttribute(splashSizes, 1));
splashGeo.setAttribute('aAlpha', new THREE.BufferAttribute(splashAlphas, 1));

const splashMaterial = new THREE.ShaderMaterial({
  uniforms: {
    uColor: { value: new THREE.Color(0xaaccdd) },
  },
  vertexShader: `
    attribute float aSize;
    attribute float aAlpha;
    varying float vAlpha;
    void main() {
      vAlpha = aAlpha;
      vec4 mvPos = modelViewMatrix * vec4(position, 1.0);
      gl_Position = projectionMatrix * mvPos;
      gl_PointSize = aSize * (150.0 / -mvPos.z);
    }
  `,
  fragmentShader: `
    uniform vec3 uColor;
    varying float vAlpha;
    void main() {
      float d = length(gl_PointCoord - 0.5);
      if (d > 0.5) discard;
      float ring = smoothstep(0.3, 0.45, d) * smoothstep(0.5, 0.45, d);
      float fill = smoothstep(0.5, 0.0, d) * 0.3;
      float alpha = (ring + fill) * vAlpha;
      gl_FragColor = vec4(uColor, alpha);
    }
  `,
  transparent: true,
  depthWrite: false,
  blending: THREE.AdditiveBlending,
});

const splashPoints = new THREE.Points(splashGeo, splashMaterial);
scene.add(splashPoints);

// Splash data
const splashes = [];

function addSplash(x, z) {
  if (splashes.length >= MAX_SPLASHES) return;
  splashes.push({
    x, y: 0.15, z,
    life: 1.0,
    maxLife: 0.4 + Math.random() * 0.3,
    size: 2.0 + Math.random() * 3.0,
    vy: 2.0 + Math.random() * 2.0,
  });
}

function updateSplashes(dt) {
  for (let i = splashes.length - 1; i >= 0; i--) {
    const s = splashes[i];
    s.life -= dt / s.maxLife;
    s.y += s.vy * dt;
    s.vy -= 8.0 * dt;
    s.size += dt * 6.0;
    if (s.life <= 0) {
      splashes.splice(i, 1);
    }
  }

  const posArr = splashGeo.attributes.position.array;
  const sizeArr = splashGeo.attributes.aSize.array;
  const alphaArr = splashGeo.attributes.aAlpha.array;

  for (let i = 0; i < MAX_SPLASHES; i++) {
    if (i < splashes.length) {
      const s = splashes[i];
      posArr[i * 3] = s.x;
      posArr[i * 3 + 1] = s.y;
      posArr[i * 3 + 2] = s.z;
      sizeArr[i] = s.size * s.life;
      alphaArr[i] = s.life * 0.7;
    } else {
      posArr[i * 3 + 1] = -100;
      sizeArr[i] = 0;
      alphaArr[i] = 0;
    }
  }

  splashGeo.attributes.position.needsUpdate = true;
  splashGeo.attributes.aSize.needsUpdate = true;
  splashGeo.attributes.aAlpha.needsUpdate = true;
}

// ============ RIPPLE MANAGEMENT ============
const ripples = [];

function addRipple(x, z, strength = 1.0) {
  ripples.push({ x, z, strength, radius: 0, speed: 12, maxRadius: 20, life: 1.0 });
}

function getIndex(ix, iz) {
  if (ix < 0 || ix > SEGMENTS || iz < 0 || iz > SEGMENTS) return -1;
  return iz * (SEGMENTS + 1) + ix;
}

// Wave simulation
const DAMPING = 0.96;
const WAVE_SPEED = 0.18;
const MAX_HEIGHT = 1.0;

function simulateWaves(dt) {
  const posArray = geometry.attributes.position.array;

  // Apply ripple sources
  for (let r = ripples.length - 1; r >= 0; r--) {
    const ripple = ripples[r];
    ripple.radius += ripple.speed * dt;
    ripple.life -= dt * 0.6;

    if (ripple.life <= 0 || ripple.radius > ripple.maxRadius) {
      ripples.splice(r, 1);
      continue;
    }

    const halfSize = SIZE / 2;
    const step = SIZE / SEGMENTS;
    const cx = Math.round((ripple.x + halfSize) / step);
    const cz = Math.round((ripple.z + halfSize) / step);
    const ringInner = Math.floor((ripple.radius - 0.5) / step);
    const ringOuter = Math.ceil((ripple.radius + 0.5) / step);

    for (let iz = Math.max(0, cz - ringOuter); iz <= Math.min(SEGMENTS, cz + ringOuter); iz++) {
      for (let ix = Math.max(0, cx - ringOuter); ix <= Math.min(SEGMENTS, cx + ringOuter); ix++) {
        const dx = ix - cx;
        const dz = iz - cz;
        const dist = Math.sqrt(dx * dx + dz * dz) * step;
        const ringDist = Math.abs(dist - ripple.radius);
        if (ringDist < 1.0) {
          const idx = getIndex(ix, iz);
          const amp = ripple.strength * ripple.life * (1.0 - ringDist) * 0.04;
          velocity[idx] += amp * Math.sin(dist * 2.0 - ripple.radius * 3.0);
        }
      }
    }
  }

  // Wave propagation
  for (let iz = 1; iz < SEGMENTS; iz++) {
    for (let ix = 1; ix < SEGMENTS; ix++) {
      const idx = getIndex(ix, iz);
      const left = getIndex(ix - 1, iz);
      const right = getIndex(ix + 1, iz);
      const up = getIndex(ix, iz - 1);
      const down = getIndex(ix, iz + 1);

      const laplacian =
        currentHeight[left] + currentHeight[right] +
        currentHeight[up] + currentHeight[down] -
        4.0 * currentHeight[idx];

      velocity[idx] += laplacian * WAVE_SPEED;
      velocity[idx] *= DAMPING;
      if (velocity[idx] > 2.0) velocity[idx] = 2.0;
      if (velocity[idx] < -2.0) velocity[idx] = -2.0;
    }
  }

  // Update heights
  for (let i = 0; i < vertexCount; i++) {
    currentHeight[i] += velocity[i];
    currentHeight[i] *= 0.99;
    if (currentHeight[i] > MAX_HEIGHT) currentHeight[i] = MAX_HEIGHT;
    if (currentHeight[i] < -MAX_HEIGHT) currentHeight[i] = -MAX_HEIGHT;
    posArray[i * 3 + 1] = currentHeight[i];
  }

  // Force edges to zero
  for (let ix = 0; ix <= SEGMENTS; ix++) {
    let idx = getIndex(ix, 0);
    currentHeight[idx] = 0; velocity[idx] = 0; posArray[idx * 3 + 1] = 0;
    idx = getIndex(ix, SEGMENTS);
    currentHeight[idx] = 0; velocity[idx] = 0; posArray[idx * 3 + 1] = 0;
  }
  for (let iz = 0; iz <= SEGMENTS; iz++) {
    let idx = getIndex(0, iz);
    currentHeight[idx] = 0; velocity[idx] = 0; posArray[idx * 3 + 1] = 0;
    idx = getIndex(SEGMENTS, iz);
    currentHeight[idx] = 0; velocity[idx] = 0; posArray[idx * 3 + 1] = 0;
  }

  geometry.attributes.position.needsUpdate = true;
  geometry.computeVertexNormals();
}

// ============ RAIN UPDATE ============
let rainIntensity = 1.0; // 0 to 1
let rainDropTimer = 0;

function updateRain(dt) {
  const posArr = rainGeo.attributes.position.array;

  const activeRainCount = Math.floor(RAIN_COUNT * rainIntensity);
  for (let i = 0; i < activeRainCount; i++) {
    posArr[i * 3 + 1] -= rainVelocities[i] * dt;

    // When raindrop hits water surface
    if (posArr[i * 3 + 1] < 0) {
      const hitX = posArr[i * 3];
      const hitZ = posArr[i * 3 + 2];

      // Create ripple at impact
      if (Math.abs(hitX) < SIZE / 2 && Math.abs(hitZ) < SIZE / 2) {
        addRipple(hitX, hitZ, 0.15 + Math.random() * 0.2);
        // Splash particles
        if (Math.random() < 0.3) {
          addSplash(hitX, hitZ);
        }
      }

      // Reset raindrop to top
      posArr[i * 3] = (Math.random() - 0.5) * RAIN_AREA;
      posArr[i * 3 + 1] = RAIN_HEIGHT_MAX + Math.random() * 10;
      posArr[i * 3 + 2] = (Math.random() - 0.5) * RAIN_AREA;
      rainVelocities[i] = 30 + Math.random() * 30;
    }
  }

  rainGeo.attributes.position.needsUpdate = true;

  // Update streaks
  const activeStreakCount = Math.floor(STREAK_COUNT * rainIntensity);
  const streakArr = streakGeo.attributes.position.array;
  for (let i = 0; i < activeStreakCount; i++) {
    const s = streakData[i];
    s.y -= s.speed * dt;

    if (s.y < 0) {
      s.x = (Math.random() - 0.5) * RAIN_AREA;
      s.y = RAIN_HEIGHT_MAX + Math.random() * 10;
      s.z = (Math.random() - 0.5) * RAIN_AREA;
    }

    const idx = i * 6;
    streakArr[idx] = s.x;
    streakArr[idx + 1] = s.y;
    streakArr[idx + 2] = s.z;
    streakArr[idx + 3] = s.x;
    streakArr[idx + 4] = s.y - s.len;
    streakArr[idx + 5] = s.z;
  }

  streakGeo.attributes.position.needsUpdate = true;
}

// Raycaster for mouse interaction
const raycaster = new THREE.Raycaster();
const mouse = new THREE.Vector2();
let isMouseDown = false;
let lastRippleTime = 0;

renderer.domElement.addEventListener('pointerdown', (e) => {
  isMouseDown = true;
  castRipple(e, 2.0);
});

renderer.domElement.addEventListener('pointermove', (e) => {
  const now = performance.now();
  if (isMouseDown) {
    if (now - lastRippleTime > 50) {
      castRipple(e, 1.8);
      lastRippleTime = now;
    }
  } else {
    if (now - lastRippleTime > 80) {
      castRipple(e, 0.4);
      lastRippleTime = now;
    }
  }
});

renderer.domElement.addEventListener('pointerup', () => {
  isMouseDown = false;
});

function castRipple(e, strength) {
  mouse.x = (e.clientX / window.innerWidth) * 2 - 1;
  mouse.y = -(e.clientY / window.innerHeight) * 2 + 1;
  raycaster.setFromCamera(mouse, camera);

  const intersects = raycaster.intersectObject(waterMesh);
  if (intersects.length > 0) {
    const p = intersects[0].point;
    addRipple(p.x, p.z, strength);
    // Add extra splashes on click
    for (let i = 0; i < 3; i++) {
      addSplash(p.x + (Math.random() - 0.5) * 2, p.z + (Math.random() - 0.5) * 2);
    }
  }
}

// ============ ANIMATION ============
const clock = new THREE.Clock();

function animate() {
  requestAnimationFrame(animate);
  const dt = Math.min(clock.getDelta(), 0.033);
  const time = clock.getElapsedTime();

  // Update rain
  updateRain(dt);

  // Update splashes
  updateSplashes(dt);

  // Simulate water
  simulateWaves(dt);

  // Update shader uniforms
  waterMaterial.uniforms.uTime.value = time;
  waterMaterial.uniforms.uCameraPos.value.copy(camera.position);

  controls.update();
  renderer.render(scene, camera);
}
animate();
