const perlin = new ImprovedNoise()
const terrainSettings = { frequency: 0.004, amplitude: 14 }

function getTerrainHeight(x, z) {
  const s = terrainSettings.frequency
  const a = terrainSettings.amplitude
  let h = 0
  h += perlin.noise(x * s, 0, z * s) * a
  h += perlin.noise(x * s * 2, 1, z * s * 2) * a * 0.5
  h += perlin.noise(x * s * 4, 2, z * s * 4) * a * 0.25
  return h
}

  const underwaterTint = vec3(0.08, 0.12, 0.1)

// Double-buffered ground update — compute ALL vertices off-screen, then swap in one frame
let _groundPending = false
let _groundTargetX = 0, _groundTargetZ = 0
let _groundRow = 0
let _groundBuffer = null // off-screen Float32Array for Z values
const GROUND_ROWS_PER_FRAME = 80 // more rows per frame since we're not uploading each tick
const _groundSegsP1 = GROUND_SEGMENTS + 1

function updateGround(px, pz) {
  _groundTargetX = px; _groundTargetZ = pz
  _groundRow = 0
  _groundPending = true
  // Pre-allocate buffer for new Z values — do NOT move mesh yet
  if (!_groundBuffer || _groundBuffer.length !== _groundSegsP1 * _groundSegsP1) {
    _groundBuffer = new Float32Array(_groundSegsP1 * _groundSegsP1)
  }
}

function tickGround() {
  if (!_groundPending) return
  const posAttr = groundGeo.attributes.position
  const endRow = Math.min(_groundSegsP1, _groundRow + GROUND_ROWS_PER_FRAME)
  const px = _groundTargetX, pz = _groundTargetZ
  // Build into off-screen buffer — no GPU upload, no mesh move
  for (let row = _groundRow; row < endRow; row++) {
    for (let col = 0; col < _groundSegsP1; col++) {
      const idx = row * _groundSegsP1 + col
      const lx = posAttr.getX(idx); const ly = posAttr.getY(idx)
      _groundBuffer[idx] = getTerrainHeight(lx + px, pz - ly) + GROUND_Y_OFFSET
    }
  }
  _groundRow = endRow
  // Only when ALL rows are done: copy buffer → geometry, move mesh, update normals — single frame swap
  if (_groundRow >= _groundSegsP1) {
    for (let i = 0; i < _groundSegsP1 * _groundSegsP1; i++) {
      posAttr.setZ(i, _groundBuffer[i])
    }
    posAttr.needsUpdate = true
    groundMesh.position.x = px; groundMesh.position.z = pz
    groundGeo.computeVertexNormals()
    _groundPending = false
  }
}

function updateGroundSync(px, pz) {
  groundMesh.position.x = px; groundMesh.position.z = pz
  const posAttr = groundGeo.attributes.position
  const total = posAttr.count
  for (let i = 0; i < total; i++) {
    const lx = posAttr.getX(i); const ly = posAttr.getY(i)
    posAttr.setZ(i, getTerrainHeight(lx + px, pz - ly) + GROUND_Y_OFFSET)
  }
  posAttr.needsUpdate = true
  groundGeo.computeVertexNormals()
}

let hfBody = null, hfCollider = null, hfCenter = { x: 0, z: 0 }

// Heightfield: build rows incrementally across frames, then swap in one go
let _hfPending = false
let _hfCx = 0, _hfCz = 0
let _hfHeights = null
let _hfRow = 0
const HF_ROWS_PER_FRAME = 20 // reduced from 40 to spread load more

function createHeightfield(cx, cz) {
  _hfCx = cx; _hfCz = cz
  _hfHeights = new Float32Array((HF_RES + 1) * (HF_RES + 1))
  _hfRow = 0
  _hfPending = true
}

// Staged finalization — split remove/create across 3 frames to avoid single-frame spike
let _hfFinalizeStage = 0
let _hfStagedHeights = null, _hfStagedCx = 0, _hfStagedCz = 0

function _finalizeHeightfield() {
  // Don't do it all at once — stage it
  _hfStagedHeights = _hfHeights
  _hfStagedCx = _hfCx; _hfStagedCz = _hfCz
  _hfFinalizeStage = 1
  _hfPending = false
  _hfHeights = null
}

function tickHeightfieldFinalize() {
  if (_hfFinalizeStage === 0) return
  if (_hfFinalizeStage === 1) {
    // Frame 1: remove old collider
    if (hfCollider) { try { physicsWorld.removeCollider(hfCollider, false) } catch(e){} hfCollider = null }
    _hfFinalizeStage = 2; return
  }
  if (_hfFinalizeStage === 2) {
    // Frame 2: remove old body
    if (hfBody) { try { physicsWorld.removeRigidBody(hfBody) } catch(e){} hfBody = null }
    _hfFinalizeStage = 3; return
  }
  if (_hfFinalizeStage === 3) {
    // Frame 3: create new body + collider
    hfBody = physicsWorld.createRigidBody(RAPIER.RigidBodyDesc.fixed().setTranslation(_hfStagedCx, 0, _hfStagedCz))
    hfCollider = physicsWorld.createCollider(
      RAPIER.ColliderDesc.heightfield(HF_RES, HF_RES, _hfStagedHeights, { x: HF_SIZE, y: 1, z: HF_SIZE }).setFriction(0.8).setRestitution(0.1), hfBody)
    hfCenter = { x: _hfStagedCx, z: _hfStagedCz }
    _hfStagedHeights = null
    _hfFinalizeStage = 0
  }
}

function tickHeightfield() {
  if (!_hfPending) return
  const endRow = Math.min(HF_RES + 1, _hfRow + HF_ROWS_PER_FRAME)
  for (let row = _hfRow; row < endRow; row++) {
    for (let col = 0; col <= HF_RES; col++) {
      const wx = _hfCx + (col / HF_RES - 0.5) * HF_SIZE
      const wz = _hfCz + (row / HF_RES - 0.5) * HF_SIZE
      _hfHeights[row + col * (HF_RES + 1)] = getTerrainHeight(wx, wz) + GROUND_Y_OFFSET
    }
  }
  _hfRow = endRow
  if (_hfRow > HF_RES) _finalizeHeightfield()
}

function createHeightfieldSync(cx, cz) {
  createHeightfield(cx, cz)
  while (_hfPending) {
    _hfRow = HF_RES + 1
    for (let row = 0; row <= HF_RES; row++) {
      for (let col = 0; col <= HF_RES; col++) {
        const wx = _hfCx + (col / HF_RES - 0.5) * HF_SIZE
        const wz = _hfCz + (row / HF_RES - 0.5) * HF_SIZE
        _hfHeights[row + col * (HF_RES + 1)] = getTerrainHeight(wx, wz) + GROUND_Y_OFFSET
      }
    }
    _finalizeHeightfield()
  }
}

function loadModelFromUploaded(name) {
  return new Promise((resolve, reject) => {
    const modelData = window.UPLOADED_3D_MODELS?.find(m => m.name === name)

  })
}

const _rockGeoCache = []
function createRockGeometry(seed) {
  if (_rockGeoCache.length < 12) {
    for (let g = 0; g < 12; g++) {
      const geo = new THREE.SphereGeometry(1, 24, 16)
      const pos = geo.attributes.position
      const freq1 = 1.5 + g * 0.3, freq2 = 2.8 + g * 0.2
      for (let i = 0; i < pos.count; i++) {
        const nx = pos.getX(i), ny = pos.getY(i), nz = pos.getZ(i)
        const len = Math.sqrt(nx * nx + ny * ny + nz * nz) || 1
        const ux = nx / len, uy = ny / len, uz = nz / len
        const n1 = Math.sin(ux * freq1 + uz * freq2 + g) * Math.cos(uy * freq1 * 1.3 + g * 0.7)
        const n2 = Math.sin(ux * freq2 * 2.1 + uy * 3.7 + g * 1.1) * 0.5
        const disp = (n1 * 0.12 + n2 * 0.06)
        const squashY = 0.45 + ((Math.sin(g * 2.1) + 1) * 0.5) * 0.4
        pos.setX(i, nx * (1 + disp))
        pos.setY(i, ny * squashY * (1 + disp * 0.5))
        pos.setZ(i, nz * (1 + disp))
      }
      geo.computeVertexNormals()
      _rockGeoCache.push(geo)
    }
  }
  return _rockGeoCache[Math.floor(((seed * 7.31) % 1 + 1) % 1 * 12)]
}

const _bushGeoCache = []
function createBushGeometry(seed) {
  if (_bushGeoCache.length < 6) {
    for (let g = 0; g < 6; g++) {
      const merged = new THREE.Group()
      const count = 3 + (g % 3)
      for (let i = 0; i < count; i++) {
        const r = 0.3 + ((Math.sin(g * 3.1 + i * 2.7) + 1) * 0.5) * 0.4
        const sphere = new THREE.Mesh(new THREE.SphereGeometry(r, 10, 8), bushMats[0])
        const ax = (Math.sin(g * 1.3 + i * 4.1) * 0.5) * 0.6
        const az = (Math.cos(g * 2.7 + i * 3.3) * 0.5) * 0.6
        sphere.position.set(ax, r * 0.6 + Math.abs(Math.sin(g + i)) * 0.15, az)
        sphere.scale.y = 0.7 + Math.abs(Math.cos(g * 0.8 + i)) * 0.3
        merged.add(sphere)
      }
      _bushGeoCache.push(merged)
    }
  }
  return _bushGeoCache[Math.floor(((seed * 5.17) % 1 + 1) % 1 * 6)].clone()
}

function createRuinColumn(seed, broken) {
  const group = new THREE.Group()
  const r = 0.25 + (seed % 0.3) * 0.5
  const fullH = 2.5 + seed * 2.5
  const h = broken ? fullH * (0.3 + ((Math.sin(seed * 47.1) + 1) * 0.5) * 0.5) : fullH

  // Column shaft
  const shaftGeo = new THREE.CylinderGeometry(r * 0.85, r, h, 10, 4)
  const pos = shaftGeo.attributes.position
  for (let i = 0; i < pos.count; i++) {
    const y = pos.getY(i)
    const angle = Math.atan2(pos.getZ(i), pos.getX(i))
    const dist = Math.sqrt(pos.getX(i)**2 + pos.getZ(i)**2) || 1
    const flute = Math.sin(angle * 8) * 0.03 * r
    const erosion = Math.sin(y * 3.1 + seed * 10) * 0.02
    pos.setX(i, pos.getX(i) + (pos.getX(i) / dist) * (flute + erosion))
    pos.setZ(i, pos.getZ(i) + (pos.getZ(i) / dist) * (flute + erosion))
  }
  shaftGeo.computeVertexNormals()
  const matIdx = Math.abs(Math.floor(seed * 30)) % 3
  const shaft = new THREE.Mesh(shaftGeo, ruinMats[matIdx])
  shaft.position.y = h / 2
  shaft.castShadow = true; shaft.receiveShadow = true
  group.add(shaft)

  // Base slab
  const baseGeo = new THREE.BoxGeometry(r * 2.8, 0.25, r * 2.8)
  const base = new THREE.Mesh(baseGeo, ruinMats[(matIdx + 1) % 3])
  base.position.y = 0.12
  base.castShadow = true; base.receiveShadow = true
  group.add(base)

  // Capital (top piece) if not too broken
  if (!broken || h > fullH * 0.6) {
    const capGeo = new THREE.BoxGeometry(r * 2.4, 0.3, r * 2.4)
    const cap = new THREE.Mesh(capGeo, ruinMats[matIdx])
    cap.position.y = h + 0.15
    cap.rotation.y = seed * 2
    cap.castShadow = true; cap.receiveShadow = true
    group.add(cap)
  }

  return group
}

function createRuinWall(seed) {
  const group = new THREE.Group()
  const w = 2.5 + seed * 3
  const fullH = 1.8 + seed * 2
  const thickness = 0.35 + seed * 0.15

  // Stacked block appearance
  const rows = Math.floor(fullH / 0.5)
  for (let row = 0; row < rows; row++) {
    const y = row * 0.5 + 0.25
    const rowSeed = Math.sin(seed * 10 + row * 3.7)
    // Random erosion: skip some blocks at top
    const missingChance = row / rows
    let x = -w / 2
    while (x < w / 2) {
      const blockW = 0.4 + Math.abs(Math.sin(x * 7 + row * 3 + seed * 20)) * 0.7
      const actualW = Math.min(blockW, w / 2 - x)
      if (actualW < 0.15) { x += actualW; continue }
      // Skip blocks randomly near top for ruined look
      if (Math.sin(x * 13.7 + row * 7.3 + seed * 50) > (1 - missingChance * 1.4)) { x += actualW + 0.05; continue }
      const blockH = 0.42 + Math.sin(x * 5 + row + seed * 8) * 0.08
      const geo = new THREE.BoxGeometry(actualW, blockH, thickness * (0.9 + Math.abs(Math.cos(x + row)) * 0.2))
      const wallMatIdx = Math.abs(Math.floor(Math.sin(x * 3 + row * 5 + seed * 7) * 100)) % 3
      const block = new THREE.Mesh(geo, ruinMats[wallMatIdx])
      block.position.set(x + actualW / 2, y, (Math.sin(x * 5 + row) * 0.03))
      block.rotation.y = Math.sin(x * 11 + seed) * 0.03
      block.castShadow = true; block.receiveShadow = true
      group.add(block)
      x += actualW + 0.05
    }
  }
  return group
}

function createRuinArchway(seed) {
  const group = new THREE.Group()
  const spacing = 1.8 + seed * 0.8
  const colH = 2.5 + seed * 1.5
  const r = 0.22

  // Two columns
  for (let side = -1; side <= 1; side += 2) {
    const col = createRuinColumn(seed + side * 0.1, false)
    col.position.x = side * spacing / 2
    group.add(col)
  }

  // Lintel across top
  const lintelGeo = new THREE.BoxGeometry(spacing + 0.8, 0.35, 0.6)
    const lintel = new THREE.Mesh(lintelGeo, ruinMats[Math.abs(Math.floor(seed * 7)) % 3])
  lintel.position.y = colH + 0.45
  lintel.rotation.z = (Math.sin(seed * 20) * 0.06)
  lintel.castShadow = true; lintel.receiveShadow = true
  group.add(lintel)

  return group
}

function createLargeRuin(seed) {
  const group = new THREE.Group()
  const type = Math.floor(seed * 100) % 4

  if (type === 0) {
    // Crumbling temple foundation — large rectangular base with partial walls and columns
    const baseW = 6 + seed * 3, baseD = 5 + seed * 2.5, baseH = 1.4
    const baseGeo = new THREE.BoxGeometry(baseW, baseH, baseD)
    const base = new THREE.Mesh(baseGeo, ruinMats[0])
    base.position.y = baseH / 2 - 0.8
    base.castShadow = true; base.receiveShadow = true
    group.add(base)

    // Step/platform
    const stepGeo = new THREE.BoxGeometry(baseW * 0.85, 0.35, baseD * 0.85)
    const step = new THREE.Mesh(stepGeo, ruinMats[1])
    step.position.y = baseH - 0.8 + 0.17
    step.castShadow = true; step.receiveShadow = true
    group.add(step)

    // Perimeter columns
    const platTop = baseH - 0.8 + 0.35
    const cols = 6 + Math.floor(seed * 4)
    for (let i = 0; i < cols; i++) {
      const angle = (i / cols) * Math.PI * 2
      const cx = Math.cos(angle) * (baseW * 0.4)
      const cz = Math.sin(angle) * (baseD * 0.4)
      const cs = (seed + i * 0.097) % 1
      const broken = Math.sin(seed * 50 + i * 11) > -0.4
      const col = createRuinColumn(cs, broken)
      col.position.set(cx, platTop, cz)
      col.rotation.x = Math.sin(seed * 7 + i) * 0.06
      col.rotation.z = Math.cos(seed * 9 + i) * 0.05
      group.add(col)
    }

    // Partial back wall
    if (seed > 0.3) {
      const wall = createRuinWall(seed * 0.7)
      wall.position.set(0, platTop, -baseD * 0.38)
      wall.scale.set(1.2, 1, 1)
      group.add(wall)
    }

    // Rubble pile — placed on the platform surface, not floating
    for (let i = 0; i < 8; i++) {
      const s = 0.2 + Math.abs(Math.sin(seed * 40 + i * 3)) * 0.6
      const geo = new THREE.BoxGeometry(s * 1.3, s * 0.5, s)
      const block = new THREE.Mesh(geo, ruinMats[i % 3])
      // Place rubble on or near the base platform
      const rx = (Math.sin(seed * 20 + i * 5) - 0.5) * baseW * 0.8
      const rz = (Math.cos(seed * 20 + i * 5) - 0.5) * baseD * 0.8
      const onPlatform = Math.abs(rx) < baseW * 0.45 && Math.abs(rz) < baseD * 0.45
      const rubbleY = onPlatform ? platTop + s * 0.25 : s * 0.25
      block.position.set(rx, rubbleY, rz)
      block.rotation.set(Math.sin(i) * 0.3, i * 1.1, Math.cos(i) * 0.2)
      block.castShadow = true; block.receiveShadow = true
      group.add(block)
    }
  } else if (type === 1) {
    // Massive broken tower
    const radius = 1.5 + seed * 0.8
    const towerH = 5 + seed * 4
    const segments = 14
    // Lower intact section
    const lowerH = towerH * (0.5 + seed * 0.3)
    const lowerGeo = new THREE.CylinderGeometry(radius * 0.9, radius, lowerH, segments, 6)
    const lPos = lowerGeo.attributes.position
    for (let i = 0; i < lPos.count; i++) {
      const y = lPos.getY(i)
      const angle = Math.atan2(lPos.getZ(i), lPos.getX(i))
      const erosion = Math.sin(angle * 5 + seed * 20) * 0.08 + Math.sin(y * 2 + seed * 10) * 0.05
      const nearTop = Math.max(0, (y - lowerH * 0.3) / (lowerH * 0.7))
      const breakage = nearTop * nearTop * Math.abs(Math.sin(angle * 7 + seed * 30)) * 0.3
      const r = Math.sqrt(lPos.getX(i) ** 2 + lPos.getZ(i) ** 2) || 1
      const factor = 1 + erosion - breakage
      lPos.setX(i, lPos.getX(i) / r * (r * factor))
      lPos.setZ(i, lPos.getZ(i) / r * (r * factor))
    }
    lowerGeo.computeVertexNormals()
    const lower = new THREE.Mesh(lowerGeo, ruinMats[0])
    lower.position.y = lowerH / 2
    lower.castShadow = true; lower.receiveShadow = true
    group.add(lower)

    // Base ring — buried foundation
    const ringGeo = new THREE.BoxGeometry(radius * 2.6, 1.2, radius * 2.6)
    const ring = new THREE.Mesh(ringGeo, ruinMats[1])
    ring.position.y = -0.2; ring.castShadow = true; ring.receiveShadow = true
    group.add(ring)

    // Fallen top section rubble — sits on ground around tower base
    for (let i = 0; i < 10; i++) {
      const s = 0.25 + Math.abs(Math.sin(seed * 50 + i * 7)) * 0.5
      const geo = new THREE.BoxGeometry(s * 1.5, s, s * 1.2)
      const piece = new THREE.Mesh(geo, ruinMats[i % 3])
      const angle = seed * 10 + i * (Math.PI * 2 / 10)
      const dist = radius + 0.5 + Math.abs(Math.sin(seed * 30 + i * 3)) * 2.5
      // Half-buried in ground
      piece.position.set(Math.cos(angle) * dist, s * 0.25, Math.sin(angle) * dist)
      piece.rotation.set(Math.sin(seed * 11 + i * 2) * 0.5, Math.sin(seed * 7 + i * 3) * 3, Math.sin(seed * 5 + i) * 0.4)
      piece.castShadow = true; piece.receiveShadow = true
      group.add(piece)
    }

    // Doorway opening — only add if tower is tall enough
    if (lowerH > 2.0) {
      const doorW = radius * 0.7, doorH = Math.min(1.8, lowerH * 0.6)
      const lGeo = new THREE.BoxGeometry(0.3, doorH, 0.4)
      const lMat = ruinMats[2]
      const lPillar = new THREE.Mesh(lGeo, lMat)
      lPillar.position.set(radius * 0.95, doorH / 2, -doorW / 2)
      lPillar.castShadow = true; group.add(lPillar)
      const rPillar = new THREE.Mesh(lGeo, lMat)
      rPillar.position.set(radius * 0.95, doorH / 2, doorW / 2)
      rPillar.castShadow = true; group.add(rPillar)
      const lintelGeo = new THREE.BoxGeometry(0.35, 0.3, doorW + 0.6)
      const lintel = new THREE.Mesh(lintelGeo, ruinMats[0])
      lintel.position.set(radius * 0.95, doorH + 0.15, 0)
      lintel.rotation.z = seed * 0.08
      lintel.castShadow = true; group.add(lintel)
    }

  } else if (type === 2) {
    // Ruined colonnade — long row of columns with partial roof
    const count = 5 + Math.floor(seed * 4)
    const spacing = 2.2
    const totalLen = count * spacing

    for (let i = 0; i < count; i++) {
      for (let side = -1; side <= 1; side += 2) {
        const cs = (seed + i * 0.11 + side * 0.05) % 1
        const broken = Math.sin(seed * 40 + i * 7 + side * 13) > -0.1
        const col = createRuinColumn(cs, broken)
        col.position.set(i * spacing - totalLen / 2, 0, side * 2.5)
        col.rotation.x = Math.sin(seed * 5 + i + side) * 0.07
        col.rotation.z = Math.cos(seed * 7 + i) * 0.05
        group.add(col)
      }
    }

    // Partial roof slabs — only connect columns that are both intact (not broken)
    for (let i = 0; i < count - 1; i++) {
      if (Math.sin(seed * 30 + i * 9) > 0.2) continue // skip some for ruined look
      // Check if both adjacent columns on at least one side are intact
      const leftBroken = Math.sin(seed * 40 + i * 7 + (-1) * 13) > -0.1
      const rightBroken = Math.sin(seed * 40 + (i+1) * 7 + (-1) * 13) > -0.1
      if (leftBroken && rightBroken) continue // both broken, skip slab
      const slabGeo = new THREE.BoxGeometry(spacing * 0.9, 0.25, 5.5)
      const slab = new THREE.Mesh(slabGeo, ruinMats[i % 3])
      slab.position.set(i * spacing - totalLen / 2 + spacing / 2, 3 + seed * 1.5, 0)
      slab.rotation.z = (Math.sin(seed * 15 + i) * 0.08)
      slab.rotation.x = (Math.cos(seed * 12 + i) * 0.04)
      slab.castShadow = true; slab.receiveShadow = true
      group.add(slab)
    }

    // Ground path / floor — buried foundation slab
    const floorGeo = new THREE.BoxGeometry(totalLen + 1, 0.8, 5.8)
    const floor = new THREE.Mesh(floorGeo, ruinMats[1])
    floor.position.y = -0.25; floor.castShadow = true; floor.receiveShadow = true
    group.add(floor)

    // Scattered rubble — on or near the floor slab
    for (let i = 0; i < 6; i++) {
      const s = 0.15 + Math.abs(Math.sin(seed * 25 + i)) * 0.45
      const geo = new THREE.BoxGeometry(s * 1.2, s * 0.6, s)
      const block = new THREE.Mesh(geo, ruinMats[i % 3])
      block.position.set(
        (Math.sin(seed * 10 + i * 4) - 0.5) * totalLen * 0.7,
        s * 0.2,
        (Math.cos(seed * 10 + i * 4) - 0.5) * 2.5
      )
      block.rotation.set(Math.sin(i) * 0.3, i * 1.5, Math.cos(i) * 0.2)
      block.castShadow = true; block.receiveShadow = true
      group.add(block)
    }

  } else {
    // Collapsed pyramid / ziggurat — with buried foundation
    const levels = 3 + Math.floor(seed * 3)
    let w = 5 + seed * 3, d = 4.5 + seed * 2.5
    // Foundation slab buried into terrain
    const foundGeo = new THREE.BoxGeometry(w + 1, 1.0, d + 1)
    const found = new THREE.Mesh(foundGeo, ruinMats[1])
    found.position.y = -0.3
    found.castShadow = true; found.receiveShadow = true
    group.add(found)
    let yOff = 0
    for (let lv = 0; lv < levels; lv++) {
      const lvH = 0.8 + (1 - lv / levels) * 0.5
      // Missing chunks
      const intact = Math.sin(seed * 20 + lv * 7) > -0.5 || lv < 2
      if (intact) {
        const geo = new THREE.BoxGeometry(w, lvH, d)
        const block = new THREE.Mesh(geo, ruinMats[lv % 3])
        block.position.y = yOff + lvH / 2
        block.rotation.y = Math.sin(seed * 3 + lv) * 0.03
        block.castShadow = true; block.receiveShadow = true
        group.add(block)
      } else {
        // Partial — half collapsed
        const halfW = w * (0.4 + Math.abs(Math.sin(seed * 15 + lv)) * 0.3)
        const geo = new THREE.BoxGeometry(halfW, lvH * 0.7, d * 0.8)
        const block = new THREE.Mesh(geo, ruinMats[lv % 3])
        block.position.set((w - halfW) * 0.3, yOff + lvH * 0.35, 0)
        block.rotation.z = seed * 0.06
        block.castShadow = true; block.receiveShadow = true
        group.add(block)
      }
      yOff += lvH
      w *= 0.78; d *= 0.78
    }

    // Staircase on one side — ascending against the pyramid wall
    const baseW2 = 5 + seed * 3
    const stairW = 1.2, stairCount = Math.min(Math.floor(levels * 2), 8)
    for (let s = 0; s < stairCount; s++) {
      const sH = 0.3
      const sGeo = new THREE.BoxGeometry(stairW, sH, 0.5)
      const step = new THREE.Mesh(sGeo, ruinMats[s % 3])
      // Steps start flush with the pyramid base and ascend along it
      step.position.set(baseW2 * 0.5 - 0.1 - s * 0.2, s * sH + sH / 2, 0)
      step.castShadow = true; step.receiveShadow = true
      if (Math.sin(seed * 30 + s * 5) > 0.6) continue // skip some broken stairs
      group.add(step)
    }

    // Rubble around base — half-buried, deterministic positions
    for (let i = 0; i < 10; i++) {
      const s = 0.2 + Math.abs(Math.sin(seed * 35 + i * 3)) * 0.7
      const geo = new THREE.BoxGeometry(s * 1.3, s * 0.5, s)
      const block = new THREE.Mesh(geo, ruinMats[i % 3])
      const angle = seed * 10 + i * (Math.PI * 2 / 10)
      const dist = 3 + Math.abs(Math.sin(seed * 20 + i * 5)) * 2.0
      // Partially buried
      block.position.set(Math.cos(angle) * dist, s * 0.1, Math.sin(angle) * dist)
      block.rotation.set(Math.sin(i) * 0.3, i * 1.2, Math.cos(i) * 0.3)
      block.castShadow = true; block.receiveShadow = true
      group.add(block)
    }
  }

  return group
}

function createRuinCluster(seed) {
  const type = seed * 100 % 3
  if (type < 1) {
    // Column cluster (2-5 columns, some broken)
    const group = new THREE.Group()
    const count = 2 + Math.floor(seed * 4)
    for (let i = 0; i < count; i++) {
      const cs = (seed + i * 0.137) % 1
      const broken = Math.sin(seed * 30 + i * 7) > -0.2
      const col = createRuinColumn(cs, broken)
      const angle = (i / count) * Math.PI * 2 + seed * 3
      const dist = 0.8 + Math.sin(seed * 5 + i) * 1.2
      col.position.x = Math.cos(angle) * dist
      col.position.z = Math.sin(angle) * dist
      col.rotation.y = seed * i * 2
      // Slight lean for age
      col.rotation.x = Math.sin(seed * 10 + i * 4) * 0.08
      col.rotation.z = Math.cos(seed * 10 + i * 4) * 0.06
      group.add(col)
    }
    // Scattered debris blocks — on ground, partially buried
    for (let i = 0; i < 3; i++) {
      const s = 0.15 + Math.abs(Math.sin(seed * 20 + i * 5)) * 0.4
      const geo = new THREE.BoxGeometry(s, s * 0.6, s * 0.8)
      const debris = new THREE.Mesh(geo, ruinMats[i % 3])
      debris.position.set(
        (Math.sin(seed * 15 + i * 8) * 2.5),
        s * 0.1,
        (Math.cos(seed * 15 + i * 8) * 2.5)
      )
      debris.rotation.set(Math.sin(seed * 3 + i * 7) * 0.5, Math.sin(seed * 5 + i * 4) * 3, Math.sin(seed * 8 + i) * 0.5)
      debris.castShadow = true; debris.receiveShadow = true
      group.add(debris)
    }
    return group
  } else if (type < 2) {
    // Wall fragment with rubble
    const group = new THREE.Group()
    const wall = createRuinWall(seed)
    group.add(wall)
    // Fallen blocks — sitting on ground, partially buried
    for (let i = 0; i < 4; i++) {
      const s = 0.2 + Math.abs(Math.sin(seed * 30 + i)) * 0.35
      const geo = new THREE.BoxGeometry(s * 1.2, s * 0.5, s)
      const block = new THREE.Mesh(geo, ruinMats[i % 3])
      block.position.set(
        (Math.sin(seed * 8 + i * 6) - 0.5) * 3,
        s * 0.1,
        0.8 + Math.abs(Math.cos(seed * 12 + i * 4)) * 1.5
      )
      block.rotation.set(0.1 * Math.sin(i), Math.sin(seed * 4 + i * 2) * 2, 0.1 * Math.cos(i))
      block.castShadow = true; block.receiveShadow = true
      group.add(block)
    }
    return group
  } else {
    return createRuinArchway(seed)
  }
}

// Debris system
const DEBRIS_COUNT = 600
const debrisPool = []
const debrisBoxGeo = new THREE.BoxGeometry(1, 1, 1)
const debrisCylGeo = new THREE.CylinderGeometry(0.5, 0.5, 1, 8)
const debrisMats = [
  new THREE.MeshStandardMaterial({ color: '#c4a96a', roughness: 0.9 }),
  new THREE.MeshStandardMaterial({ color: '#b89a5e', roughness: 0.92 }),
  new THREE.MeshStandardMaterial({ color: '#8a7d6b', roughness: 0.88 }),
  new THREE.MeshStandardMaterial({ color: '#a88c52', roughness: 0.95 }),
]
for (let i = 0; i < DEBRIS_COUNT; i++) {
  debrisPool.push({
    mesh: null, life: 0, maxLife: 0,
    vx: 0, vy: 0, vz: 0, rx: 0, ry: 0, rz: 0,
    gravity: 9.8, drag: 0.97,
  })
}
let debrisIndex = 0
const debrisGroup = new THREE.Group()
scene.add(debrisGroup)

function spawnDebris(wx, wy, wz, impactVx, impactVy, impactVz, count, scale, matIndex, isRuin = false) {
  for (let n = 0; n < count; n++) {
    const idx = debrisIndex % DEBRIS_COUNT; debrisIndex++
    const d = debrisPool[idx]
    if (d.mesh) { debrisGroup.remove(d.mesh); d.mesh = null }
    const sx = (0.08 + Math.random() * 0.18) * scale
    const sy = (0.05 + Math.random() * 0.12) * scale
    const sz = (0.06 + Math.random() * 0.15) * scale
    const geo = isRuin ? debrisCylGeo : debrisBoxGeo
    const m = new THREE.Mesh(geo, debrisMats[(matIndex + n) % debrisMats.length])
    m.scale.set(sx, sy, sz)
    m.position.set(
      wx + (Math.random() - 0.5) * scale * 0.8,
      wy + Math.random() * scale * 0.5,
      wz + (Math.random() - 0.5) * scale * 0.8
    )
    m.rotation.set(Math.random() * 6, Math.random() * 6, Math.random() * 6)
    m.castShadow = true
    debrisGroup.add(m)
    d.mesh = m
    const spread = 3 + scale * 1.5
    d.vx = impactVx * 0.3 + (Math.random() - 0.5) * spread
    d.vy = 2 + Math.random() * (3 + scale)
    d.vz = impactVz * 0.3 + (Math.random() - 0.5) * spread
    d.rx = (Math.random() - 0.5) * 8
    d.ry = (Math.random() - 0.5) * 8
    d.rz = (Math.random() - 0.5) * 8
    d.maxLife = 1.5 + Math.random() * 2.5
    d.life = d.maxLife
    d.gravity = 8 + Math.random() * 4
    d.drag = 0.96 + Math.random() * 0.03
  }
}

function updateDebris(dt) {
  for (let i = 0; i < DEBRIS_COUNT; i++) {
    const d = debrisPool[i]
    if (d.life <= 0 || !d.mesh) continue
    d.life -= dt
    if (d.life <= 0) {
      debrisGroup.remove(d.mesh)
      d.mesh = null
      continue
    }
    d.vy -= d.gravity * dt
    d.vx *= d.drag; d.vz *= d.drag
    d.mesh.position.x += d.vx * dt
    d.mesh.position.y += d.vy * dt
    d.mesh.position.z += d.vz * dt
    d.mesh.rotation.x += d.rx * dt
    d.mesh.rotation.y += d.ry * dt
    d.mesh.rotation.z += d.rz * dt
    const t = d.life / d.maxLife
    if (t < 0.3) d.mesh.scale.multiplyScalar(0.97)
    // Simple floor check — use car Y as approximate ground instead of noise per debris
    const approxGround = chassisPos.y - 3
    if (d.mesh.position.y < approxGround) {
      d.mesh.position.y = approxGround
      d.vy *= -0.3
      d.vx *= 0.7; d.vz *= 0.7
      d.rx *= 0.5; d.ry *= 0.5; d.rz *= 0.5
    }
  }
}

// Solid (non-destructible) colliders for large ruins
const solidColliders = new Map()

function addSolidCollider(mesh, cellKey, halfW, halfH, halfD) {
  const wp = new THREE.Vector3()
  mesh.getWorldPosition(wp)
  const body = physicsWorld.createRigidBody(
    RAPIER.RigidBodyDesc.fixed().setTranslation(wp.x, wp.y + halfH, wp.z)
  )
  const collider = physicsWorld.createCollider(
    RAPIER.ColliderDesc.cuboid(halfW, halfH, halfD)
      .setFriction(0.8)
      .setSensor(false),
    body
  )
  solidColliders.set(collider.handle, { mesh, body, collider, cellKey })
}

function addDestructibleCollider(mesh, cellKey, halfW, halfH, halfD, isRuin) {
  const wp = new THREE.Vector3()
  mesh.getWorldPosition(wp)
  const body = physicsWorld.createRigidBody(
    RAPIER.RigidBodyDesc.fixed().setTranslation(wp.x, wp.y + halfH, wp.z)
  )
  const collider = physicsWorld.createCollider(
    RAPIER.ColliderDesc.cuboid(halfW, halfH, halfD)
      .setActiveEvents(RAPIER.ActiveEvents.COLLISION_EVENTS)
      .setFriction(0.8)
      .setSensor(false),
    body
  )
  destructibleMap.set(collider.handle, { mesh, body, collider, cellKey, isRuin, halfW, halfH, halfD })
}

// Track all shared/cached geometries to never dispose them
const _sharedGeos = new Set([..._rockGeoCache, debrisBoxGeo, debrisCylGeo])

function destroyObject(handle, impactVel) {
  const info = destructibleMap.get(handle)
  if (!info) return
  destructibleMap.delete(handle)
  const { mesh, body, collider, cellKey, isRuin, halfW, halfH, halfD } = info
  const wp = new THREE.Vector3()
  try { mesh.getWorldPosition(wp) } catch (e) { wp.copy(mesh.position) }
  const scale = Math.max(halfW, halfH, halfD) * 2
  const debrisCount = Math.min(25, Math.max(6, Math.floor(scale * 4)))
  spawnDebris(wp.x, wp.y, wp.z, impactVel.x, impactVel.y, impactVel.z, debrisCount, scale, isRuin ? 0 : 2, isRuin)
  // Remove mesh from scene
  if (mesh.parent) mesh.parent.remove(mesh)
  mesh.traverse(c => {
    if (c.geometry && !_sharedGeos.has(c.geometry) && !_rockGeoCache.includes(c.geometry)) {
      c.geometry.dispose()
    }
  })
  // Remove physics safely
  try { physicsWorld.removeCollider(collider, false) } catch (e) {}
  try { physicsWorld.removeRigidBody(body) } catch (e) {}
  // Remove from cell tracking
  const cellObjs = scatterCells.get(cellKey)
  if (cellObjs) {
    const idx = cellObjs.indexOf(mesh)
    if (idx !== -1) cellObjs.splice(idx, 1)
  }
}

function seededRand(x, z, seed) {
  let n = Math.sin(x * 12.9898 + z * 78.233 + seed * 43.1234) * 43758.5453
  return n - Math.floor(n)
}

function buildCell(cellX, cellZ) {
  const key = cellX + ',' + cellZ
  if (scatterCells.has(key)) return
  const objs = []
  const wx0 = cellX * SCATTER_CELL, wz0 = cellZ * SCATTER_CELL
  const r1 = seededRand(wx0, wz0, 1)
  const r2 = seededRand(wx0, wz0, 2)

  if (r1 < 0.28) {
    const wx = wx0 + (seededRand(wx0, wz0, 3) - 0.5) * SCATTER_CELL * 0.8
    const wz = wz0 + (seededRand(wx0, wz0, 4) - 0.5) * SCATTER_CELL * 0.8
    const h = getTerrainHeight(wx, wz) + GROUND_Y_OFFSET
    if (h >= biomeSettings.waterLevel - 0.5) {
      const scale = 0.3 + seededRand(wx0, wz0, 5) * 1.2
      const geo = createRockGeometry(r1 + cellX * 0.137 + cellZ * 0.293)
      const mat = rockMats[Math.floor(seededRand(wx0, wz0, 6) * 3)]
      const mesh = new THREE.Mesh(geo, mat)
      mesh.position.set(wx, h - 0.15 * scale, wz)
      mesh.rotation.set(
        seededRand(wx0, wz0, 7) * 0.4,
        seededRand(wx0, wz0, 8) * Math.PI * 2,
        seededRand(wx0, wz0, 9) * 0.3
      )
      mesh.scale.set(scale, scale * (0.5 + seededRand(wx0, wz0, 10) * 0.6), scale)
      mesh.castShadow = true; mesh.receiveShadow = true
      scatterGroup.add(mesh); objs.push(mesh)
    }
  }

  if (r2 < 0.22) {
    const wx = wx0 + (seededRand(wx0, wz0, 11) - 0.5) * SCATTER_CELL * 0.8
    const wz = wz0 + (seededRand(wx0, wz0, 12) - 0.5) * SCATTER_CELL * 0.8
    const h = getTerrainHeight(wx, wz) + GROUND_Y_OFFSET
    if (h >= biomeSettings.sandEnd - 0.5) {
      const scale = 0.5 + seededRand(wx0, wz0, 13) * 0.8
      const bg = createBushGeometry(seededRand(wx0, wz0, 16))
      const mat = bushMats[Math.floor(seededRand(wx0, wz0, 14) * 3)]
      bg.traverse(c => { if (c.isMesh) { c.material = mat; c.castShadow = true; c.receiveShadow = true } })
      bg.position.set(wx, h - 0.05, wz)
      bg.scale.setScalar(scale)
      bg.rotation.y = seededRand(wx0, wz0, 15) * Math.PI * 2
      scatterGroup.add(bg); objs.push(bg)
    }
  }

  // Ancient ruins — spawn on sandy areas, less frequent
  const r3 = seededRand(wx0, wz0, 20)
  if (r3 < 0.04) {
    const wx = wx0 + (seededRand(wx0, wz0, 21) - 0.5) * SCATTER_CELL * 0.6
    const wz = wz0 + (seededRand(wx0, wz0, 22) - 0.5) * SCATTER_CELL * 0.6
    const h = getTerrainHeight(wx, wz) + GROUND_Y_OFFSET
    // Only place on sandy/low terrain
    if (h >= biomeSettings.waterLevel + 0.5 && h <= biomeSettings.dirtEnd + 1) {
      const ruinSeed = seededRand(wx0, wz0, 23)
      const ruin = createRuinCluster(ruinSeed)
      const scale = 0.8 + seededRand(wx0, wz0, 24) * 0.6
      ruin.position.set(wx, h - 0.1, wz)
      ruin.scale.setScalar(scale)
      ruin.rotation.y = seededRand(wx0, wz0, 25) * Math.PI * 2
      scatterGroup.add(ruin); objs.push(ruin)
      addDestructibleCollider(ruin, key, 1.8 * scale, 1.5 * scale, 1.8 * scale, true)
    }
  }

  // Large building-sized ruins — very rare
  const r4 = seededRand(wx0, wz0, 30)
  if (r4 < 0.012) {
    const wx = wx0 + (seededRand(wx0, wz0, 31) - 0.5) * SCATTER_CELL * 0.4
    const wz = wz0 + (seededRand(wx0, wz0, 32) - 0.5) * SCATTER_CELL * 0.4
    const h = getTerrainHeight(wx, wz) + GROUND_Y_OFFSET
    if (h >= biomeSettings.waterLevel + 1.0 && h <= biomeSettings.dirtEnd + 2) {
      const bigSeed = seededRand(wx0, wz0, 33)
      const bigRuin = createLargeRuin(bigSeed)
      const scale = 2.2 + seededRand(wx0, wz0, 34) * 1.3
      bigRuin.position.set(wx, h - 0.6 * scale, wz)
      bigRuin.scale.setScalar(scale)
      bigRuin.rotation.y = seededRand(wx0, wz0, 35) * Math.PI * 2
      scatterGroup.add(bigRuin); objs.push(bigRuin)
      addSolidCollider(bigRuin, key, 4.5 * scale, 3 * scale, 4 * scale)
    }
  }

  // Add colliders for rocks
  if (r1 < 0.28 && objs.length > 0) {
    const rockMesh = objs[0]
    if (rockMesh && rockMesh.isMesh) {
      const s = rockMesh.scale
      addDestructibleCollider(rockMesh, key, s.x * 0.6, s.y * 0.5, s.z * 0.6, false)
    }
  }

  scatterCells.set(key, objs)
}

function removeCell(key) {
  const objs = scatterCells.get(key)
  if (!objs) return
  for (const obj of objs) {
    scatterGroup.remove(obj)
    obj.traverse(c => {
      if (c.geometry && !_rockGeoCache.includes(c.geometry) && !_sharedGeos?.has(c.geometry) && !_bushGeoCache.some(bg => { let found = false; bg.traverse(bc => { if (bc.geometry === c.geometry) found = true }); return found })) {
        c.geometry.dispose()
      }
    })
  }
  // Collect handles to remove, then remove outside iteration
  const destroyHandles = []
  const solidHandles = []
  for (const [handle, info] of destructibleMap) {
    if (info.cellKey === key) destroyHandles.push(handle)
  }
  for (const [handle, info] of solidColliders) {
    if (info.cellKey === key) solidHandles.push(handle)
  }
  for (const handle of destroyHandles) {
    const info = destructibleMap.get(handle)
    if (info) {
      try { physicsWorld.removeCollider(info.collider, false) } catch (e) {}
      try { physicsWorld.removeRigidBody(info.body) } catch (e) {}
      destructibleMap.delete(handle)
    }
  }
  for (const handle of solidHandles) {
    const info = solidColliders.get(handle)
    if (info) {
      try { physicsWorld.removeCollider(info.collider, false) } catch (e) {}
      try { physicsWorld.removeRigidBody(info.body) } catch (e) {}
      solidColliders.delete(handle)
    }
  }
  scatterCells.delete(key)
}

let _lastScatterCX = null, _lastScatterCZ = null
let _scatterBuildQueue = []
const SCATTER_CELLS_PER_FRAME = 2
const MAX_SCATTER_CELLS = 800

function updateScatter(px, pz) {
  const cx = Math.round(px / SCATTER_CELL)
  const cz = Math.round(pz / SCATTER_CELL)

  // Skip if player hasn't moved to a new cell
  if (cx === _lastScatterCX && cz === _lastScatterCZ) return
  _lastScatterCX = cx; _lastScatterCZ = cz

  // Remove far-away cells first — remove MORE per update to keep count bounded
  const maxDist = SCATTER_RANGE + 2
  let removed = 0
  for (const key of [...scatterCells.keys()]) {
    if (removed >= 12) break
    const sep = key.indexOf(',')
    const kx = parseInt(key.substring(0, sep))
    const kz = parseInt(key.substring(sep + 1))
    if (Math.abs(kx - cx) > maxDist || Math.abs(kz - cz) > maxDist) {
      removeCell(key)
      removed++
    }
  }

  // Queue cells to build (sorted by distance so nearest load first)
  // Only queue if we're under the cell cap
  if (scatterCells.size < MAX_SCATTER_CELLS) {
    _scatterBuildQueue = []
    for (let gx = -SCATTER_RANGE; gx <= SCATTER_RANGE; gx++) {
      for (let gz = -SCATTER_RANGE; gz <= SCATTER_RANGE; gz++) {
        const key = (cx + gx) + ',' + (cz + gz)
        if (!scatterCells.has(key)) {
          _scatterBuildQueue.push({ cx: cx + gx, cz: cz + gz, d: gx * gx + gz * gz })
        }
      }
    }
    _scatterBuildQueue.sort((a, b) => a.d - b.d)
    // Cap queue length to avoid massive backlogs
    if (_scatterBuildQueue.length > 80) _scatterBuildQueue.length = 80
  }
}

function tickScatterBuild() {
  if (scatterCells.size >= MAX_SCATTER_CELLS) return
  const count = Math.min(SCATTER_CELLS_PER_FRAME, _scatterBuildQueue.length)
  for (let i = 0; i < count; i++) {
    const item = _scatterBuildQueue.shift()
    // Double-check cell wasn't already built while queued
    const key = item.cx + ',' + item.cz
    if (scatterCells.has(key)) continue
    buildCell(item.cx, item.cz)
  }
}

updateScatter(0, 0)
// Finish initial scatter synchronously
while (_scatterBuildQueue.length > 0) tickScatterBuild()

let windDustIndex = 0
let windTime = 0

// Cache terrain heights for wind dust to avoid per-particle noise lookups
let _windHeightCache = 0, _windHeightCacheX = -99999, _windHeightCacheZ = -99999
function spawnWindDust(cx, cy, cz) {
  const i = windDustIndex % WIND_DUST_COUNT; windDustIndex++
  const d = windDustData[i]
  const angle = Math.random() * Math.PI * 2
  const dist = Math.random() * windSettings.spawnRadius
  d.x = cx + Math.cos(angle) * dist
  d.z = cz + Math.sin(angle) * dist
  // Use cached height from car pos instead of per-particle noise
  d.y = cy + 0.5 + Math.random() * windSettings.spawnHeight
  d.vx = windSettings.dirX * (0.6 + Math.random() * 0.4)
  d.vy = (Math.random() - 0.4) * 0.5
  d.vz = windSettings.dirZ * (0.6 + Math.random() * 0.4)
  d.maxLife = windSettings.lifetime + Math.random() * windSettings.lifetimeVariance
  d.life = d.maxLife
  d.size = windSettings.particleSize + Math.random() * windSettings.sizeVariance
  d.opacity = windSettings.opacity * (0.3 + Math.random() * 0.7)
  d.wobblePhase = Math.random() * Math.PI * 2
}

function updateWindDust(dt, cx, cy, cz) {
  windTime += dt
  const gustX = Math.sin(windTime * 0.7) * windSettings.gustStrength
  const gustZ = Math.cos(windTime * 0.5) * windSettings.gustStrength * 0.6
  
  for (let i = 0; i < WIND_DUST_COUNT; i++) {
    const d = windDustData[i]
    if (d.life <= 0) {
      _dustMat4.makeScale(0, 0, 0)
      windDustIMesh.setMatrixAt(i, _dustMat4)
      windDustColors[i*3] = 0; windDustColors[i*3+1] = 0; windDustColors[i*3+2] = 0
      continue
    }
    d.life -= dt
    const turbX = Math.sin(windTime * 3.1 + d.wobblePhase) * windSettings.turbulence
    const turbZ = Math.cos(windTime * 2.7 + d.wobblePhase * 1.3) * windSettings.turbulence * 0.8
    const wobble = Math.sin(windTime * d.wobbleSpeed + d.wobblePhase) * d.wobbleAmp
    d.vx += (gustX + turbX) * dt
    d.vy += wobble * dt * 0.5
    d.vz += (gustZ + turbZ) * dt
    d.vx *= windSettings.drag; d.vy *= windSettings.drag; d.vz *= windSettings.drag
    d.x += d.vx * dt; d.y += d.vy * dt; d.z += d.vz * dt
    
    // Simple floor clamp — skip expensive per-particle terrain lookup
    if (d.y < biomeSettings.waterLevel + 0.3) d.y = biomeSettings.waterLevel + 0.3 + Math.random() * 0.5
    
    const t = Math.max(0, d.life / d.maxLife)
    const fadeIn = Math.min(1, (d.maxLife - d.life) * 3)
    const fadeOut = t < 0.3 ? t / 0.3 : 1
    const alpha = fadeIn * fadeOut
    const s = (d.size * 0.08) * alpha
    
    const distFromCam = Math.sqrt((d.x - cx) ** 2 + (d.z - cz) ** 2)
    const distFade = Math.max(0, 1 - distFromCam / (windSettings.spawnRadius * 1.2))
    
    windDustColors[i*3] = 0.76
    windDustColors[i*3+1] = 0.62
    windDustColors[i*3+2] = 0.38
    
    _dustPos.set(d.x, d.y, d.z)
    _dustScale.set(s * distFade, s * distFade, s * distFade)
    _dustMat4.compose(_dustPos, camera.quaternion, _dustScale)
    windDustIMesh.setMatrixAt(i, _dustMat4)
  }
  windDustIMesh.instanceMatrix.needsUpdate = true
  windDustIMesh.instanceColor.needsUpdate = true
}

let splashIndex = 0

  for (let n = 0; n < count; n++) {
    const i = splashIndex % SPLASH_COUNT; splashIndex++
    const d = splashData[i]
    d.x = wx + (Math.random() - 0.5) * splashSettings.spread
    d.y = wy + Math.random() * 0.15
    d.z = wz + (Math.random() - 0.5) * splashSettings.spread
    const angle = Math.random() * Math.PI * 2
    const outSpeed = splashSettings.velocitySpread * (0.5 + Math.random() * 0.5)
    d.vx = vx * 0.1 + Math.cos(angle) * outSpeed
    d.vy = splashSettings.upwardForce * (0.5 + Math.random() * 0.5)
    d.vz = vz * 0.1 + Math.sin(angle) * outSpeed
    d.gravity = 6 + Math.random() * 4
    d.maxLife = splashSettings.lifetime + Math.random() * splashSettings.lifetimeVariance
    d.life = d.maxLife; d.size = 0.08 + Math.random() * 0.15
    d.opacity = splashSettings.opacity * (0.6 + Math.random() * 0.4)
  }
}

function updateSplashParticles(dt) {
  for (let i = 0; i < SPLASH_COUNT; i++) {
    const d = splashData[i]
    if (d.life <= 0) { _dustMat4.makeScale(0, 0, 0); splashIMesh.setMatrixAt(i, _dustMat4); splashColors[i*3]=0; splashColors[i*3+1]=0; splashColors[i*3+2]=0; continue }
    d.life -= dt; d.x += d.vx * dt; d.vy -= d.gravity * dt; d.y += d.vy * dt; d.z += d.vz * dt
    d.vx *= splashSettings.drag; d.vz *= splashSettings.drag
    const t = Math.max(0, d.life / d.maxLife)
    const smoothT = t * t * (3 - 2 * t)
    const grow = d.size * (1 + (1 - t) * 0.8) * smoothT
    const white = 0.3 * (1 - t)
    splashColors[i*3] = Math.min(1, splashSettings.colorR + white)
    splashColors[i*3+1] = Math.min(1, splashSettings.colorG + white)
    splashColors[i*3+2] = Math.min(1, splashSettings.colorB + white * 0.5)
    _dustPos.set(d.x, d.y, d.z); _dustScale.set(grow, grow, grow)
    _dustMat4.compose(_dustPos, camera.quaternion, _dustScale)
    splashIMesh.setMatrixAt(i, _dustMat4)
  }
  splashIMesh.instanceMatrix.needsUpdate = true; splashIMesh.instanceColor.needsUpdate = true
}

let dustIndex = 0
function emitDust(wx, wy, wz, vx, vy, vz, count) {
  const c = count * 3
  for (let n = 0; n < c; n++) {
    const i = dustIndex % DUST_COUNT; dustIndex++
    const d = dustData[i]
    d.x = wx + (Math.random() - 0.5) * dustSettings.spread
    d.y = wy - WHEEL_R * 0.8 + Math.random() * 0.1
    d.z = wz + (Math.random() - 0.5) * dustSettings.spread
    d.vx = vx * 0.15 + (Math.random() - 0.5) * dustSettings.velocitySpread * 0.5
    d.vy = Math.random() * dustSettings.upwardForce * 0.5 + 0.25
    d.vz = vz * 0.15 + (Math.random() - 0.5) * dustSettings.velocitySpread * 0.5
    d.maxLife = dustSettings.lifetime + Math.random() * dustSettings.lifetimeVariance
    d.life = d.maxLife; d.size = 0.15 + Math.random() * 0.25
    d.opacity = dustSettings.opacity * (0.7 + Math.random() * 0.3)
  }
}

function updateDustParticles(dt) {
  for (let i = 0; i < DUST_COUNT; i++) {
    const d = dustData[i]
    if (d.life <= 0) { _dustMat4.makeScale(0, 0, 0); dustIMesh.setMatrixAt(i, _dustMat4); dustColors[i*3]=0; dustColors[i*3+1]=0; dustColors[i*3+2]=0; continue }
    d.life -= dt; d.x += d.vx * dt; d.y += d.vy * dt; d.z += d.vz * dt
    d.vx *= dustSettings.drag; d.vy *= (dustSettings.drag - 0.01); d.vz *= dustSettings.drag
    const t = Math.max(0, d.life / d.maxLife)
    const smoothT = t * t * (3 - 2 * t)
    const grow = d.size * (1 + (1 - t) * 1.5) * smoothT
    const colorT = 1 - t
    dustColors[i*3] = dustSettings.colorR + (dustSettings.colorR2 - dustSettings.colorR) * colorT
    dustColors[i*3+1] = dustSettings.colorG + (dustSettings.colorG2 - dustSettings.colorG) * colorT
    dustColors[i*3+2] = dustSettings.colorB + (dustSettings.colorB2 - dustSettings.colorB) * colorT
    _dustPos.set(d.x, d.y, d.z); _dustScale.set(grow, grow, grow)
    _dustMat4.compose(_dustPos, camera.quaternion, _dustScale)
    dustIMesh.setMatrixAt(i, _dustMat4)
  }
  dustIMesh.instanceMatrix.needsUpdate = true; dustIMesh.instanceColor.needsUpdate = true
}

let lastGroundX = 0, lastGroundZ = 0
let jumpState = 'ready', jumpTimer = 0, prevSpace = false

function resetCar() {
  const h = getTerrainHeight(0, 0) + 3
  chassisBody.setTranslation({ x: 0, y: h, z: 0 }, true)
  chassisBody.setRotation({ x: 0, y: 0, z: 0, w: 1 }, true)
  chassisBody.setLinvel({ x: 0, y: 0, z: 0 }, true)
  chassisBody.setAngvel({ x: 0, y: 0, z: 0 }, true)
  wheelSpins.fill(0)
}

async function animate() {
  try {
  const accel = (keys.KeyW ? 1 : 0) - (keys.KeyS ? 1 : 0)
  const steer = (keys.KeyA ? 1 : 0) - (keys.KeyD ? 1 : 0)
  const spaceDown = !!keys.Space
  const jumpPressed = spaceDown && !prevSpace
  prevSpace = spaceDown
  const boosting = !!keys.ShiftLeft || !!keys.ShiftRight
  if (keys.KeyR) { resetCar(); keys.KeyR = false }

  const vel = chassisBody.linvel()
  const fwd = tmpV.set(1, 0, 0).applyQuaternion(chassisQuat)
  const forwardSpeed = fwd.x * vel.x + fwd.y * vel.y + fwd.z * vel.z
  const speed = Math.sqrt(vel.x ** 2 + vel.y ** 2 + vel.z ** 2)

  const topSpeed = carSettings.maxSpeed * (boosting ? carSettings.boostMultiplier : 1)
  const overSpeed = Math.max(0, Math.abs(forwardSpeed) - topSpeed)
  let engineForce = (accel * carSettings.acceleration * (boosting ? carSettings.boostMultiplier : 1)) / (1 + overSpeed)
  let brake = 0
  if (!accel) brake = carSettings.deceleration * 0.15
  if (speed > 0.5 && ((accel > 0 && forwardSpeed < -0.5) || (accel < 0 && forwardSpeed > 0.5))) { brake = carSettings.deceleration; engineForce = 0 }

  const chassisUp = tmpV.set(0, 1, 0).applyQuaternion(chassisQuat)
  const isUpsideDown = chassisUp.y < Math.cos((84 * Math.PI) / 180)

  if (isUpsideDown && jumpPressed) {
    const mass = chassisBody.mass()
    chassisBody.applyImpulse({ x: 0, y: carSettings.flipForce * mass, z: 0 }, true)
    const torque = tmpV.set(1, 0, 0).applyQuaternion(chassisQuat).multiplyScalar(carSettings.flipForce * 0.66 * mass)
    chassisBody.applyTorqueImpulse({ x: torque.x, y: torque.y, z: torque.z }, true)
    jumpState = 'cooldown'; jumpTimer = 0
  }

  let wheelsOnGround = 0
  for (let i = 0; i < 4; i++) if (vehicle.wheelIsInContact(i)) wheelsOnGround++
  if (!isUpsideDown && jumpState === 'ready' && jumpPressed && wheelsOnGround >= 2) { jumpState = 'crouching'; jumpTimer = 0 }

  if (jumpState === 'crouching') {
    jumpTimer += FIXED_DT
    for (let i = 0; i < 4; i++) { vehicle.setWheelSuspensionRestLength(i, 0.05); vehicle.setWheelSuspensionStiffness(i, 80) }
    if (jumpTimer >= carSettings.jumpCrouchTime) {
      for (let i = 0; i < 4; i++) { vehicle.setWheelSuspensionRestLength(i, SUSP_REST); vehicle.setWheelSuspensionStiffness(i, 12) }
      const up = tmpV.set(0, 1, 0).applyQuaternion(chassisQuat)
      const mass = chassisBody.mass()
      chassisBody.applyImpulse({ x: up.x * carSettings.jumpForce * mass, y: up.y * carSettings.jumpForce * mass, z: up.z * carSettings.jumpForce * mass }, true)
      jumpState = 'cooldown'; jumpTimer = 0
    }
  }
  if (jumpState === 'cooldown') { jumpTimer += FIXED_DT; if (jumpTimer > 0.3 && wheelsOnGround >= 2) jumpState = 'ready' }

  const steerAngle = steer * carSettings.steer * Math.sqrt(carSettings.acceleration / 5)
  vehicle.setWheelSteering(0, steerAngle); vehicle.setWheelSteering(1, steerAngle)
  for (let i = 0; i < 4; i++) { vehicle.setWheelEngineForce(i, engineForce); vehicle.setWheelBrake(i, brake); vehicle.setWheelFrictionSlip(i, carSettings.grip) }

  if (chassisPos.y < biomeSettings.waterLevel) {
    const v = chassisBody.linvel(); const drag = 0.92
    chassisBody.setLinvel({ x: v.x * drag, y: v.y * drag, z: v.z * drag }, true)
  }

  try { vehicle.updateVehicle(FIXED_DT) } catch (e) {}
  try { physicsWorld.step() } catch (e) { console.warn('Physics step failed:', e) }

  const p = chassisBody.translation(); const r = chassisBody.rotation()
  chassisPos.set(p.x, p.y, p.z); chassisQuat.set(r.x, r.y, r.z, r.w)
  chassisGroup.position.copy(chassisPos); chassisGroup.quaternion.copy(chassisQuat)

  for (let i = 0; i < 4; i++) {
    const conn = wheelConns[i]
    const inContact = vehicle.wheelIsInContact(i)
    const suspLen = inContact ? (vehicle.wheelSuspensionLength(i) ?? SUSP_REST) : SUSP_REST
    tmpV.set(conn.x, conn.y - suspLen, conn.z).applyQuaternion(chassisQuat).add(chassisPos)
    wheelDebugMeshes[i].position.copy(tmpV)
    wheelMeshes[i].position.x = tmpV.x
    wheelMeshes[i].position.y = THREE.MathUtils.lerp(wheelMeshes[i].position.y, tmpV.y, carSettings.tireLerp)
    wheelMeshes[i].position.z = tmpV.z
    tmpQ.copy(chassisQuat)
    if (i < 2) tmpQ.multiply(tmpQ2.setFromAxisAngle(UP, steerAngle))
    if (inContact) wheelSpins[i] -= (forwardSpeed / WHEEL_R) * FIXED_DT
    tmpQ.multiply(tmpQ2.setFromAxisAngle(SPIN_AXIS, wheelSpins[i]))
    wheelDebugMeshes[i].quaternion.copy(tmpQ)
    wheelMeshes[i].quaternion.slerp(tmpQ, carSettings.tireLerp)
  }

  const snapX = Math.round(chassisPos.x / GROUND_SNAP) * GROUND_SNAP
  const snapZ = Math.round(chassisPos.z / GROUND_SNAP) * GROUND_SNAP
  if (snapX !== lastGroundX || snapZ !== lastGroundZ) { lastGroundX = snapX; lastGroundZ = snapZ; updateGround(snapX, snapZ) }

  // Incremental updates — spread across frames
  tickGround()
  tickHeightfield()
  tickHeightfieldFinalize()
  tickScatterBuild()

  // Check if scatter needs new cells — every 20 frames
  if (!animate._scatterFrame) animate._scatterFrame = 0
  if (++animate._scatterFrame % 20 === 0) {
    updateScatter(chassisPos.x, chassisPos.z)
    // Also do extra cleanup if cell count is getting high
    if (scatterCells.size > MAX_SCATTER_CELLS * 0.9) {
      const cx = Math.round(chassisPos.x / SCATTER_CELL)
      const cz = Math.round(chassisPos.z / SCATTER_CELL)
      const maxDist = SCATTER_RANGE + 2
      let cleaned = 0
      for (const key of [...scatterCells.keys()]) {
        if (cleaned >= 20) break
        const sep = key.indexOf(',')
        const kx = parseInt(key.substring(0, sep))
        const kz = parseInt(key.substring(sep + 1))
        if (Math.abs(kx - cx) > maxDist || Math.abs(kz - cz) > maxDist) {
          removeCell(key); cleaned++
        }
      }
    }
  }

  waterMesh.position.x = chassisPos.x; waterMesh.position.z = chassisPos.z

  const dxHF = chassisPos.x - hfCenter.x; const dzHF = chassisPos.z - hfCenter.z
  if (!_hfPending && _hfFinalizeStage === 0 && dxHF * dxHF + dzHF * dzHF > HF_RETHRESHOLD * HF_RETHRESHOLD) createHeightfield(chassisPos.x, chassisPos.z)

  const azRad = (lightSettings.azimuth * Math.PI) / 180
  const elRad = (lightSettings.elevation * Math.PI) / 180
  const lightDist = 30
  dirLight.position.set(
    chassisPos.x + Math.cos(elRad) * Math.sin(azRad) * lightDist,
    chassisPos.y + Math.sin(elRad) * lightDist,
    chassisPos.z + Math.cos(elRad) * Math.cos(azRad) * lightDist)
  dirLight.target.position.copy(chassisPos); dirLight.target.updateMatrixWorld()

  if (carSettings.debug) updateDebug()

  if (carSettings.orbitControls) { orbit.target.lerp(chassisPos, 0.1); orbit.update() }
  else {
    const camFwd = tmpV.set(1, 0, 0).applyQuaternion(chassisQuat)
    cameraIdeal.copy(chassisPos).addScaledVector(camFwd, -cameraSettings.distance)
    cameraIdeal.y = chassisPos.y + cameraSettings.height
    camera.position.lerp(cameraIdeal, cameraSettings.smoothing)
    camera.lookAt(chassisPos.x, chassisPos.y + cameraSettings.lookHeight, chassisPos.z)
  }

  const groundSpeed = Math.sqrt(vel.x * vel.x + vel.z * vel.z)
  const inWater = chassisPos.y < biomeSettings.waterLevel + 1.5
  if (dustSettings.enabled && wheelsOnGround >= 2 && groundSpeed > dustSettings.minSpeed && !inWater) {
    const emitCount = Math.min(3, Math.floor(groundSpeed * dustSettings.emitRate))
    for (let i = 0; i < 4; i++) if (vehicle.wheelIsInContact(i)) { const wp = wheelMeshes[i].position; emitDust(wp.x, wp.y, wp.z, -vel.x, 0, -vel.z, emitCount) }
  }
  updateDustParticles(FIXED_DT)
  updateDebris(FIXED_DT)

  // Check collisions with destructible objects — limit to 2 destructions per frame, skip if no destructibles nearby
  const cVel = chassisBody.linvel()
  const cSpeed = Math.sqrt(cVel.x ** 2 + cVel.y ** 2 + cVel.z ** 2)
  if (cSpeed > 3 && destructibleMap.size > 0) {
    const toDestroy = []
    try {
      const chassisCol = chassisBody.collider(0)
      if (chassisCol) {
        physicsWorld.contactPairsWith(chassisCol, (otherCollider) => {
          if (otherCollider && toDestroy.length < 2) {
            const handle = otherCollider.handle
            if (destructibleMap.has(handle)) {
              toDestroy.push(handle)
            }
          }
        })
      }
    } catch (e) { /* physics query failed, skip */ }
    for (const handle of toDestroy) {
      try { destroyObject(handle, { x: cVel.x, y: cVel.y, z: cVel.z }) } catch (e) {}
    }
  }

  if (splashSettings.enabled && inWater && groundSpeed > splashSettings.minSpeed) {
    const sc = Math.min(5, Math.floor(groundSpeed * splashSettings.emitRate))
    for (let i = 0; i < 4; i++) { const wp = wheelMeshes[i].position; if (wp.y < biomeSettings.waterLevel + 0.5) emitSplash(wp.x, biomeSettings.waterLevel, wp.z, -vel.x, 0, -vel.z, sc) }
  }
  updateSplashParticles(FIXED_DT)

  // Spawn and update ambient wind dust
  const windSpawnRate = 3
  for (let s = 0; s < windSpawnRate; s++) {
    spawnWindDust(chassisPos.x, chassisPos.y, chassisPos.z)
  }
  updateWindDust(FIXED_DT, camera.position.x, camera.position.y, camera.position.z)

  if (chassisPos.y < -30) resetCar()

  renderer.render(scene, camera)
  stats.update()
  try { await renderer.resolveTimestampsAsync('render'); await renderer.resolveTimestampsAsync('compute') } catch (e) {}
  } catch (err) { console.error('Animation loop error:', err) }
}

setLoading('Ready', 100)
const loaderEl = document.getElementById('loader')
if (loaderEl) { setTimeout(() => { loaderEl.style.opacity = '0'; setTimeout(() => loaderEl.remove(), 600) }, 300) }

renderer.setAnimationLoop(animate)

  camera.aspect = innerWidth / innerHeight; camera.updateProjectionMatrix()
  renderer.setPixelRatio(Math.min(devicePixelRatio, 2)); renderer.setSize(innerWidth, innerHeight)
