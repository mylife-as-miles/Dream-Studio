function playSnapSound() {
  if (audioMuted) return;
  const now = audioCtx.currentTime;
  // Layer 1: soft click
  const clickOsc = audioCtx.createOscillator();
  const clickGain = audioCtx.createGain();
  clickOsc.type = 'sine';
  clickOsc.frequency.setValueAtTime(1400, now);
  clickOsc.frequency.exponentialRampToValueAtTime(500, now + 0.025);
  clickGain.gain.setValueAtTime(0.045, now);
  clickGain.gain.exponentialRampToValueAtTime(0.001, now + 0.04);
  clickOsc.connect(clickGain).connect(audioCtx.destination);
  clickOsc.start(now);
  clickOsc.stop(now + 0.04);

  // Layer 2: gentle thud
  const thudOsc = audioCtx.createOscillator();
  const thudGain = audioCtx.createGain();
  thudOsc.type = 'sine';
  thudOsc.frequency.setValueAtTime(240, now);
  thudOsc.frequency.exponentialRampToValueAtTime(100, now + 0.06);
  thudGain.gain.setValueAtTime(0.055, now);
  thudGain.gain.exponentialRampToValueAtTime(0.001, now + 0.07);
  thudOsc.connect(thudGain).connect(audioCtx.destination);
  thudOsc.start(now);
  thudOsc.stop(now + 0.07);

  // Layer 3: subtle noise texture
  const bufferSize = audioCtx.sampleRate * 0.02;
  const noiseBuffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
  const noiseData = noiseBuffer.getChannelData(0);
  for (let i = 0; i < bufferSize; i++) noiseData[i] = (Math.random() * 2 - 1) * 0.25;
  const noiseSrc = audioCtx.createBufferSource();
  const noiseGain = audioCtx.createGain();
  const noiseFilter = audioCtx.createBiquadFilter();
  noiseFilter.type = 'bandpass';
  noiseFilter.frequency.value = 2500;
  noiseFilter.Q.value = 1.5;
  noiseSrc.buffer = noiseBuffer;
  noiseGain.gain.setValueAtTime(0.025, now);
  noiseGain.gain.exponentialRampToValueAtTime(0.001, now + 0.025);
  noiseSrc.connect(noiseFilter).connect(noiseGain).connect(audioCtx.destination);
  noiseSrc.start(now);
}

function playRemoveSound() {
  if (audioMuted) return;
  const now = audioCtx.currentTime;
  // Gentle reverse pop
  const osc = audioCtx.createOscillator();
  const gain = audioCtx.createGain();
  osc.type = 'sine';
  osc.frequency.setValueAtTime(180, now);
  osc.frequency.exponentialRampToValueAtTime(700, now + 0.05);
  gain.gain.setValueAtTime(0.035, now);
  gain.gain.exponentialRampToValueAtTime(0.001, now + 0.06);
  osc.connect(gain).connect(audioCtx.destination);
  osc.start(now);
  osc.stop(now + 0.06);

  // Whisper air release
  const bufferSize = audioCtx.sampleRate * 0.03;
  const buf = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
  const d = buf.getChannelData(0);
  for (let i = 0; i < bufferSize; i++) d[i] = (Math.random() * 2 - 1) * 0.5;
  const src = audioCtx.createBufferSource();
  const g = audioCtx.createGain();
  const f = audioCtx.createBiquadFilter();
  f.type = 'highpass';
  f.frequency.value = 2500;
  src.buffer = buf;
  g.gain.setValueAtTime(0.02, now);
  g.gain.exponentialRampToValueAtTime(0.001, now + 0.035);
  src.connect(f).connect(g).connect(audioCtx.destination);
  src.start(now);
}

// ─── Procedural textures ───
function createCanvas(w, h) {
  const c = document.createElement('canvas');
  c.width = w; c.height = h;
  return { canvas: c, ctx: c.getContext('2d') };
}

function seededRandom(seed) {
  let s = seed;
  return function () {
    s = (s * 16807 + 0) % 2147483647;
    return (s - 1) / 2147483646;
  };
}

function generateNormalMap() {
  const size = 256;
  const { canvas, ctx } = createCanvas(size, size);
  const imageData = ctx.createImageData(size, size);
  const data = imageData.data;
  const rand = seededRandom(42);
  for (let i = 0; i < size * size; i++) {
    let nx = 128 + (rand() - 0.5) * 20;
    let ny = 128 + (rand() - 0.5) * 20;
    data[i * 4] = Math.max(0, Math.min(255, nx));
    data[i * 4 + 1] = Math.max(0, Math.min(255, ny));
    data[i * 4 + 2] = 255;
    data[i * 4 + 3] = 255;
  }
  ctx.putImageData(imageData, 0, 0);
  const tex = new THREE.CanvasTexture(canvas);
  tex.wrapS = tex.wrapT = THREE.RepeatWrapping;
  return tex;
}

function generateClearcoatNormalMap(opts = {}) {
  const scratchCount = opts.count ?? 200;
  const scratchStrength = opts.strength ?? 60;
  const scratchWidth = opts.width ?? 2;
  const minLen = opts.minLength ?? 30;
  const maxLen = opts.maxLength ?? 300;
  const curvature = opts.curvature ?? 1.5;
  const seed = opts.seed ?? 254;
  const size = 256;
  const { canvas, ctx } = createCanvas(size, size);
  ctx.fillStyle = 'rgb(128,128,255)';
  ctx.fillRect(0, 0, size, size);
  const imageData = ctx.getImageData(0, 0, size, size);
  const data = imageData.data;
  const floatR = new Float32Array(size * size).fill(128);
  const floatG = new Float32Array(size * size).fill(128);
  const rand = seededRandom(seed);
  for (let s = 0; s < scratchCount; s++) {
    const startX = rand() * size, startY = rand() * size;
    const angle = rand() * Math.PI * 2;
    const len = minLen + rand() * (maxLen - minLen);
    const curve = (rand() - 0.5) * curvature;
    const steps = Math.ceil(len);
    for (let i = 0; i < steps; i++) {
      const t = i / steps;
      const curAngle = angle + curve * t;
      const px = startX + Math.cos(curAngle) * i;
      const py = startY + Math.sin(curAngle) * i;
      const perpX = -Math.sin(curAngle), perpY = Math.cos(curAngle);
      const endFade = Math.min(t * 6, 1.0) * Math.min((1 - t) * 6, 1.0);
      for (let tt = -scratchWidth; tt <= scratchWidth; tt++) {
        const ox = ((Math.round(px + perpX * tt * 0.5) % size) + size) % size;
        const oy = ((Math.round(py + perpY * tt * 0.5) % size) + size) % size;
        const fi = oy * size + ox;
        const normalizedPos = tt / scratchWidth;
        const side = normalizedPos < 0 ? -1 : normalizedPos > 0 ? 1 : 0;
        const strength = scratchStrength * normalizedPos * normalizedPos * endFade;
        floatR[fi] += perpX * side * strength;
        floatG[fi] += perpY * side * strength;
      }
    }
  }
  for (let i = 0; i < size * size; i++) {
    data[i * 4] = Math.max(0, Math.min(255, Math.round(floatR[i])));
    data[i * 4 + 1] = Math.max(0, Math.min(255, Math.round(floatG[i])));
  }
  ctx.putImageData(imageData, 0, 0);
  const tex = new THREE.CanvasTexture(canvas);
  tex.wrapS = tex.wrapT = THREE.RepeatWrapping;
  return tex;
}

// Rotate equirectangular HDR pixels by horizontal shift (yaw) and generate new PMREM
function rotateEquirectPixels(rawHdr, yawDeg, tiltDeg) {
  const w = rawHdr.image.width;
  const h = rawHdr.image.height;
  const srcData = rawHdr.image.data;
  const channels = srcData.length / (w * h); // 4 for RGBA half-float
  const dstData = new srcData.constructor(srcData.length);

  const yawShift = ((yawDeg % 360) + 360) % 360;
  const pixelShift = Math.round((yawShift / 360) * w);
  // Tilt: vertical pixel shift (approximate for small angles)
  const tiltShift = Math.round((tiltDeg / 180) * h);

  for (let y = 0; y < h; y++) {
    const srcY = Math.min(Math.max(y + tiltShift, 0), h - 1);
    for (let x = 0; x < w; x++) {
      const srcX = (x + pixelShift) % w;
      const dstIdx = (y * w + x) * channels;
      const srcIdx = (srcY * w + srcX) * channels;
      for (let c = 0; c < channels; c++) {
        dstData[dstIdx + c] = srcData[srcIdx + c];
      }
    }
  }

  const rotatedTex = new THREE.DataTexture(dstData, w, h, rawHdr.format, rawHdr.type);
  rotatedTex.mapping = THREE.EquirectangularReflectionMapping;
  rotatedTex.minFilter = THREE.LinearFilter;
  rotatedTex.magFilter = THREE.LinearFilter;
  rotatedTex.flipY = rawHdr.flipY !== undefined ? rawHdr.flipY : true;
  rotatedTex.needsUpdate = true;
  return rotatedTex;
}

function applyHdrWithRotation(rawHdr, yawDeg, tiltDeg) {
  if (!rawHdr) return;
  // Rotate the raw HDR pixels
  const rotated = rotateEquirectPixels(rawHdr, yawDeg, tiltDeg);
  const envMap = pmremGenerator.fromEquirectangular(rotated).texture;
  rotated.dispose();
  scene.environment = envMap;
  scene.background = envMap;
}

function loadHDR(index) {
  currentHdrIndex = index;
  const entry = HDR_ENVIRONMENTS[index];

  // Update active button
  document.querySelectorAll('.hdr-btn').forEach((btn, i) => {
    btn.classList.toggle('active', i === index);
  });

  if (hdrCache[entry.url]) {
    currentRawHdr = hdrCache[entry.url].raw;
    applyHdrWithRotation(currentRawHdr, envRotationDeg, envTiltDeg);
    return;
  }

  rgbeLoader.load(entry.url, (hdr) => {
    // Store the raw equirect HDR (before PMREM) for future rotation
    const rawCloneData = new hdr.image.data.constructor(hdr.image.data);
    const rawClone = new THREE.DataTexture(rawCloneData, hdr.image.width, hdr.image.height, hdr.format, hdr.type);
    rawClone.mapping = THREE.EquirectangularReflectionMapping;
    rawClone.minFilter = THREE.LinearFilter;
    rawClone.magFilter = THREE.LinearFilter;
    rawClone.flipY = hdr.flipY;
    rawClone.needsUpdate = true;

    const envMap = pmremGenerator.fromEquirectangular(hdr).texture;
    hdr.dispose();
    hdrCache[entry.url] = { raw: rawClone, pmrem: envMap };
    if (currentHdrIndex === index) {
      currentRawHdr = rawClone;
      applyHdrWithRotation(currentRawHdr, envRotationDeg, envTiltDeg);
    }
  }, undefined, () => {});
}

// ─── Roughness variation map (worn/grip areas on studs) ───
function generateRoughnessVariationMap() {
  const size = 256;
  const { canvas, ctx } = createCanvas(size, size);
  ctx.fillStyle = 'rgb(128,128,128)';
  ctx.fillRect(0, 0, size, size);
  const imageData = ctx.getImageData(0, 0, size, size);
  const data = imageData.data;
  const rand = seededRandom(99);
  // Broad low-frequency worn patches (simulates finger grip areas)
  for (let p = 0; p < 30; p++) {
    const cx = rand() * size, cy = rand() * size;
    const radius = 40 + rand() * 120;
    const strength = 20 + rand() * 35;
    for (let y = Math.max(0, Math.floor(cy - radius)); y < Math.min(size, Math.ceil(cy + radius)); y++) {
      for (let x = Math.max(0, Math.floor(cx - radius)); x < Math.min(size, Math.ceil(cx + radius)); x++) {
        const dx = x - cx, dy = y - cy;
        const dist = Math.sqrt(dx * dx + dy * dy);
        if (dist < radius) {
          const falloff = 1 - (dist / radius);
          const smooth = falloff * falloff * (3 - 2 * falloff);
          const idx = (y * size + x) * 4;
          // Lower value = shinier (less rough)
          data[idx] = Math.max(0, data[idx] - strength * smooth);
          data[idx + 1] = data[idx];
          data[idx + 2] = data[idx];
        }
      }
    }
  }
  // Fine grain noise overlay
  for (let i = 0; i < size * size; i++) {
    const noise = (rand() - 0.5) * 16;
    data[i * 4] = Math.max(0, Math.min(255, data[i * 4] + noise));
    data[i * 4 + 1] = data[i * 4];
    data[i * 4 + 2] = data[i * 4];
  }
  ctx.putImageData(imageData, 0, 0);
  const tex = new THREE.CanvasTexture(canvas);
  tex.wrapS = tex.wrapT = THREE.RepeatWrapping;
  return tex;
}

// ─── Build LEGO brick geometry ───
function normalizeGeo(geo, keepSmooth) {
  let g;
  if (keepSmooth && geo.index) {
    g = geo.toNonIndexed();
  } else {
    g = geo.index ? geo.toNonIndexed() : geo.clone();
    g.computeVertexNormals();
  }
  const pos = g.getAttribute('position');
  const nor = g.getAttribute('normal');
  let uvArr = g.getAttribute('uv');
  if (!uvArr) {
    uvArr = new THREE.BufferAttribute(new Float32Array(pos.count * 2), 2);
    g.setAttribute('uv', uvArr);
  }
  const newGeo = new THREE.BufferGeometry();
  newGeo.setAttribute('position', pos);
  newGeo.setAttribute('normal', nor);
  newGeo.setAttribute('uv', uvArr);
  return newGeo;
}

// Cache geometries by type key
const geoCache = {};

function buildBrickGeometry(cols, rows) {
  const key = `${cols}x${rows}`;
  if (geoCache[key]) return geoCache[key];

  const geometries = [];
  const brickW = cols * UNIT, brickD = rows * UNIT;

  const outerBox = new RoundedBoxGeometry(brickW, BRICK_HEIGHT, brickD, 2, 0.05);
  outerBox.translate(0, BRICK_HEIGHT / 2, 0);
  geometries.push(normalizeGeo(outerBox, true));

  const studPts = [];
  studPts.push(new THREE.Vector2(STUD_RADIUS, 0));
  studPts.push(new THREE.Vector2(STUD_RADIUS, STUD_HEIGHT - STUD_BEVEL));
  const bevelSegs = 2;
  for (let i = 0; i <= bevelSegs; i++) {
    const a = (i / bevelSegs) * (Math.PI / 2);
    studPts.push(new THREE.Vector2(
      STUD_RADIUS - STUD_BEVEL + Math.cos(a) * STUD_BEVEL,
      STUD_HEIGHT - STUD_BEVEL + Math.sin(a) * STUD_BEVEL
    ));
  }
  studPts.push(new THREE.Vector2(0, STUD_HEIGHT));

  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      const sx = -brickW / 2 + UNIT / 2 + c * UNIT;
      const sz = -brickD / 2 + UNIT / 2 + r * UNIT;
      const studGeo = new THREE.LatheGeometry(studPts, SEGMENTS);
      studGeo.translate(sx, BRICK_HEIGHT, sz);
      geometries.push(normalizeGeo(studGeo, true));
    }
  }

  const merged = mergeGeometries(geometries, false);
  geoCache[key] = merged || new THREE.BoxGeometry(brickW, BRICK_HEIGHT, brickD);
  return geoCache[key];
}

// ─── Material factory ───
const materialCache = {};

function getBrickMaterial(colorHex, opacity = 1.0) {
  const key = colorHex + '_' + opacity;
  if (materialCache[key]) return materialCache[key];
  const c = new THREE.Color(colorHex);
  const luminance = 0.2126 * c.r + 0.7152 * c.g + 0.0722 * c.b;
  // Three tiers: very light (white), medium-light (pastels), normal
  const isVeryLight = luminance > 0.7;
  const isLight = luminance > 0.45;

  let roughness, clearcoat, clearcoatRoughness, normalStr, envInt, ccNormalStr, sheenStr;
  if (isVeryLight) {
    roughness = 0.65;
    clearcoat = 0.04;
    clearcoatRoughness = 0.5;
    normalStr = 0.06;
    envInt = 0.08;
    ccNormalStr = 0.04;
    sheenStr = 0.08;
  } else if (isLight) {
    roughness = 0.32;
    clearcoat = 0.35;
    clearcoatRoughness = 0.22;
    normalStr = 0.14;
    envInt = 0.5;
    ccNormalStr = 0.07;
    sheenStr = 0.3;
  } else {
    roughness = 0.22;
    clearcoat = 0.5;
    clearcoatRoughness = 0.15;
    normalStr = 0.23;
    envInt = 0.8;
    ccNormalStr = 0.14;
    sheenStr = 0.35;
  }

  const isGhost = opacity < 1;
  // Compute a lighter, desaturated sheen tint from the base color for fresnel edge brightening
  const sheenC = c.clone();
  const hslObj = { h: 0, s: 0, l: 0 };
  sheenC.getHSL(hslObj);
  sheenC.setHSL(hslObj.h, hslObj.s * 0.3, Math.min(1, hslObj.l + 0.45));

  const mat = new THREE.MeshPhysicalMaterial({
    color: c,
    roughness,
    metalness: 0.0,
    clearcoat: isGhost ? 0 : clearcoat,
    clearcoatRoughness: isGhost ? 0 : clearcoatRoughness,
    normalMap: isGhost ? null : normalMap,
    normalScale: new THREE.Vector2(normalStr, normalStr),
    clearcoatNormalMap: isGhost ? null : clearcoatNormalMap,
    clearcoatNormalScale: new THREE.Vector2(ccNormalStr, ccNormalStr),
    roughnessMap: isGhost ? null : roughnessVariationMap,
    envMapIntensity: isGhost ? 0.2 : envInt,
    sheen: isGhost ? 0 : sheenStr,
    sheenRoughness: isGhost ? 0.5 : 0.4,
    sheenColor: isGhost ? new THREE.Color(1, 1, 1) : sheenC,
    transparent: isGhost,
    opacity,
    side: isGhost ? THREE.DoubleSide : THREE.FrontSide,
    depthWrite: !isGhost,
    polygonOffset: isGhost,
    polygonOffsetFactor: isGhost ? 1 : 0,
    polygonOffsetUnits: isGhost ? 1 : 0,
  });
  materialCache[key] = mat;
  return mat;
}

const ssrMaxDistU = uniform(1.60);

// ─── Screen-Space Reflections via TSL ───
// We'll build an SSR node that raymarches in screen space using depth + normals.
// Strategy: for each pixel, reconstruct view-space position from depth,
// reflect the view direction around the normal, then march through screen space
// to find where the reflected ray hits geometry.

const projMatU = uniform(camera.projectionMatrix);
const projInvMatU = uniform(camera.projectionMatrixInverse);

// TSL function: reconstruct view-space position from screen UV + depth texture
const ssrNode = Fn(([colorTex, depthTex, normalTex]) => {
  const uv = screenUV;
  const baseColor = colorTex.sample(uv);
  const rawDepth = depthTex.sample(uv).x;

  // Early out: sky pixels (depth ≈ 1) — return base color
  const isSky = rawDepth.greaterThanEqual(0.999);

  // Extract projection matrix components
  const A = projMatU.element(2).element(2);
  const B = projMatU.element(3).element(2);

  // Linearize depth: NDC z from [0,1] depth buffer → view-space linear Z (positive)
  const ndcZ = rawDepth.mul(2.0).sub(1.0);
  const linZ = B.div(ndcZ.add(A));

  // Reconstruct clip-space XY from screen UV
  // screenUV.y is top-down (0=top, 1=bottom) but clip Y is bottom-up, so flip Y
  const clipX = uv.x.mul(2.0).sub(1.0);
  const clipY = float(1.0).sub(uv.y).mul(2.0).sub(1.0);

  // Reconstruct view position: unproject using inverse projection diagonal
  const vx = clipX.mul(projInvMatU.element(0).element(0)).mul(linZ);
  const vy = clipY.mul(projInvMatU.element(1).element(1)).mul(linZ);
  const viewPos = vec3(vx, vy, linZ.negate());

  // Read view-space normal directly from MRT (transformedNormalView outputs [-1,1] already)
  const normN = normalTex.sample(uv).xyz.normalize();

  // View direction (from camera origin to fragment, in view space)
  const viewDir = viewPos.normalize();

  // Reflect the view direction around the surface normal
  const reflDirNorm = viewDir.sub(normN.mul(viewDir.dot(normN).mul(2.0))).normalize();

  // Fresnel: stronger reflections at grazing angles
  const NdotV = normN.dot(viewDir.negate()).clamp(0.0, 1.0);
  const fresnel = float(1.0).sub(NdotV).pow(ssrFresnelPowU).clamp(0.0, 1.0);

  // Early out: if reflection points toward camera (z > 0), skip — it won't hit anything on screen
  const reflZ = reflDirNorm.z;

  // Raymarch in view space, project each sample to screen space
  const hitColor = vec3(0.0, 0.0, 0.0).toVar();
  const hitWeight = float(0.0).toVar();

  // Track the ray march state for binary refinement
  const hitT = float(0.0).toVar();
  const prevT = float(0.0).toVar();

  // Only raymarch if reflection goes into the scene (z < 0 in view space)
  If(reflZ.lessThan(0.1).and(isSky.not()), () => {
    // Coarse linear march: find the first intersection interval
    Loop(32, ({ i }) => {
      const fi = float(i).add(1.0);
      const t = fi.div(32.0).mul(ssrMaxDistU);
      const samplePos = viewPos.add(reflDirNorm.mul(t));

      const negZ = samplePos.z.negate();
      const sClipX = samplePos.x.mul(projMatU.element(0).element(0)).div(negZ);
      const sClipY = samplePos.y.mul(projMatU.element(1).element(1)).div(negZ);

      const sUV = vec2(
        sClipX.mul(0.5).add(0.5),
        float(1.0).sub(sClipY.mul(0.5).add(0.5))
      );

      const inBounds = sUV.x.greaterThanEqual(0.0)
        .and(sUV.x.lessThanEqual(1.0))
        .and(sUV.y.greaterThanEqual(0.0))
        .and(sUV.y.lessThanEqual(1.0));

      If(inBounds.and(hitWeight.lessThan(0.5)), () => {
        const sampledDepth = depthTex.sample(sUV).x;
        const sampledNdcZ = sampledDepth.mul(2.0).sub(1.0);
        const sampledLinZ = B.div(sampledNdcZ.add(A));

        const rayZ = negZ;
        const diff = rayZ.sub(sampledLinZ);

        const isHit = diff.greaterThan(0.0).and(diff.lessThan(ssrThicknessU));
        const notSky = sampledDepth.lessThan(0.999);

        If(isHit.and(notSky), () => {
          hitT.assign(t);
          hitWeight.assign(1.0);
        });
      });

      // Track previous t for binary refinement range
      If(hitWeight.lessThan(0.5), () => {
        prevT.assign(t);
      });
    });

    // Binary search refinement: narrow down the exact hit point between prevT and hitT
    If(hitWeight.greaterThan(0.5), () => {
      const loT = prevT.toVar();
      const hiT = hitT.toVar();

      Loop(6, () => {
        const midT = loT.add(hiT).mul(0.5);
        const midPos = viewPos.add(reflDirNorm.mul(midT));

        const midNegZ = midPos.z.negate();
        const midClipX = midPos.x.mul(projMatU.element(0).element(0)).div(midNegZ);
        const midClipY = midPos.y.mul(projMatU.element(1).element(1)).div(midNegZ);

        const midUV = vec2(
          midClipX.mul(0.5).add(0.5),
          float(1.0).sub(midClipY.mul(0.5).add(0.5))
        );

        const midDepth = depthTex.sample(midUV).x;
        const midNdcZ = midDepth.mul(2.0).sub(1.0);
        const midLinZ = B.div(midNdcZ.add(A));

        const midDiff = midNegZ.sub(midLinZ);

        // If ray is behind surface, move hi down; otherwise move lo up
        If(midDiff.greaterThan(0.0), () => {
          hiT.assign(midT);
        }).Else(() => {
          loT.assign(midT);
        });
      });

      // Sample the refined hit point
      const finalT = loT.add(hiT).mul(0.5);
      const finalPos = viewPos.add(reflDirNorm.mul(finalT));

      const finalNegZ = finalPos.z.negate();
      const finalClipX = finalPos.x.mul(projMatU.element(0).element(0)).div(finalNegZ);
      const finalClipY = finalPos.y.mul(projMatU.element(1).element(1)).div(finalNegZ);

      const finalUV = vec2(
        finalClipX.mul(0.5).add(0.5),
        float(1.0).sub(finalClipY.mul(0.5).add(0.5))
      );

      // Edge fade
      const edgeX = finalUV.x.mul(float(1.0).sub(finalUV.x)).mul(4.0).clamp(0.0, 1.0);
      const edgeY = finalUV.y.mul(float(1.0).sub(finalUV.y)).mul(4.0).clamp(0.0, 1.0);
      const edgeFade = edgeX.mul(edgeY);

      // Distance fade
      const distFade = float(1.0).sub(finalT.div(ssrMaxDistU)).clamp(0.0, 1.0);

      const sampledColor = colorTex.sample(finalUV).xyz;
      hitColor.assign(sampledColor.mul(edgeFade).mul(distFade));
    });
  });

  // Combine: blend reflection onto base color weighted by fresnel and strength
  const reflectionMix = hitWeight.mul(fresnel).mul(ssrStrengthU).mul(ssrFadeU);
  const finalColor = baseColor.xyz.add(hitColor.mul(reflectionMix));

  // When SSR is disabled or it's sky, return base color
  const result = isSky.select(baseColor.xyz, ssrEnabledU.greaterThan(0.5).select(finalColor, baseColor.xyz));

  return vec4(result, baseColor.w);
});

const ssrOutput = ssrNode(scenePassColor, scenePassDepth, scenePassNormal);

// ─── Bloom ───
let bloomEnabled = true;
const bloomStrengthU = uniform(0.2);
const bloomRadiusU = uniform(0.2);
const bloomThresholdU = uniform(1.6);

// ─── Depth of Field ───
let dofEnabled = false;
const focusDistU = uniform(7.0);
const focalLengthU = uniform(2.0);
const bokehScaleU = uniform(15.0);

function rebuildPostPipeline() {
  // Build source: AO → SSR → Bloom → DOF
  let source = ssrOutput;

  // Mix AO onto SSR output
  source = source.mul(vec4(aoMixed, 1.0));

  if (bloomEnabled) {
    const bloomOnly = bloom(source, bloomStrengthU, bloomRadiusU, bloomThresholdU);
    source = source.add(bloomOnly);
  }

  if (dofEnabled) {
    const sceneViewZ = scenePass.getViewZNode();
    source = dof(source, sceneViewZ, focusDistU, focalLengthU, bokehScaleU);
  }

  // Rebuild the ENTIRE output chain so all nodes reference the correct source
  const _uv = screenUV.mul(vec2(float(window.innerWidth), float(window.innerHeight)));
  const _seed = _uv.add(vec2(time.mul(7.23), time.mul(3.91)));
  const _noise = hash(_seed.dot(vec2(127.1, 311.7)));
  const _grainAmount = _noise.sub(0.5).mul(grainIntensityU).mul(grainEnabledU);

  const _shoulderThreshold = float(0.68).sub(highlightClampU.mul(0.25));
  const _compressed = source.sub(_shoulderThreshold).div(float(1.0).sub(_shoulderThreshold)).mul(float(1.0).sub(_shoulderThreshold).mul(float(1.0).sub(highlightClampU.mul(0.55)))).add(_shoulderThreshold);
  const _shouldApply = source.greaterThan(_shoulderThreshold);
  const _clamped = vec3(
    _shouldApply.select(_compressed.x, source.x),
    _shouldApply.select(_compressed.y, source.y),
    _shouldApply.select(_compressed.z, source.z),
  );
  const _highlightMixed = highlightClampU.greaterThan(0.001).select(_clamped, source);
  const _withGrain = clamp(_highlightMixed.add(_grainAmount), 0.0, 1.0);

  postProcessing.outputNode = _withGrain;
  postProcessing.needsUpdate = true;
}

// Build initial pipeline (DOF off)

function positionGhostBrick() {
  if (!ghostBrick) return;
  const type = BRICK_TYPES[selectedTypeIndex];
  const cols = rotated ? type.rows : type.cols;
  const rows = rotated ? type.cols : type.rows;

  // Calculate layer height
  const layer = getLayerAtPosition(ghostGridX, ghostGridZ, cols, rows);

  const x = -baseplateSize / 2 + ghostGridX * UNIT + (cols * UNIT) / 2;
  const z = -baseplateSize / 2 + ghostGridZ * UNIT + (rows * UNIT) / 2;
  const y = layer * BRICK_HEIGHT;

  ghostBrick.position.set(x, y, z);

  // Check if placement is valid
  const valid = isValidPlacement(ghostGridX, ghostGridZ, cols, rows, layer);
  ghostBrick.material = getBrickMaterial(valid ? BRICK_COLORS[selectedColorIndex].hex : '#ff0000', 0.45);
}

function getLayerAtPosition(gx, gz, cols, rows) {
  let maxLayer = 0;
  for (const b of placedBricks) {
    // Check overlap
    const bCols = b.cols, bRows = b.rows;
    if (gx < b.gridX + bCols && gx + cols > b.gridX && gz < b.gridZ + bRows && gz + rows > b.gridZ) {
      maxLayer = Math.max(maxLayer, b.layer + 1);
    }
  }
  return maxLayer;
}

function isValidPlacement(gx, gz, cols, rows, layer) {
  // Bounds check
  if (gx < 0 || gz < 0 || gx + cols > GRID_SIZE || gz + rows > GRID_SIZE) return false;

  // Check for exact overlap at same layer
  for (const b of placedBricks) {
    if (b.layer !== layer) continue;
    if (gx < b.gridX + b.cols && gx + cols > b.gridX && gz < b.gridZ + b.rows && gz + rows > b.gridZ) {
      return false;
    }
  }
  return true;
}

function isBrickDroppingAtPosition(gx, gz, cols, rows) {
  for (const b of placedBricks) {
    if (!b.mesh.userData.dropping) continue;
    // Check if any cell overlaps
    if (gx < b.gridX + b.cols && gx + cols > b.gridX && gz < b.gridZ + b.rows && gz + rows > b.gridZ) {
      return true;
    }
  }
  return false;
}

// ─── Raycasting for mouse position ───
const raycaster = new THREE.Raycaster();
const mouse = new THREE.Vector2();
const baseplatePlane = new THREE.Plane(new THREE.Vector3(0, 1, 0), 0);

let prevGhostGridX = -1, prevGhostGridZ = -1;

// Throttle mouse move for raycast performance
let mouseMoveRafPending = false;
let lastMouseEvent = null;

function onMouseMoveThrottled(event) {
  lastMouseEvent = event;
  if (!mouseMoveRafPending) {
    mouseMoveRafPending = true;
    requestAnimationFrame(() => {
      mouseMoveRafPending = false;
      if (lastMouseEvent) onMouseMove(lastMouseEvent);
    });
  }
}

// Reusable vector to avoid allocation per frame
const _intersectPoint = new THREE.Vector3();

function onMouseMove(event) {
  const rect = renderer.domElement.getBoundingClientRect();
  mouse.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
  mouse.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;

  if (!placementMode) return;

  raycaster.setFromCamera(mouse, camera);
  const hit = raycaster.ray.intersectPlane(baseplatePlane, _intersectPoint);

  if (hit) {
    const type = BRICK_TYPES[selectedTypeIndex];
    const cols = rotated ? type.rows : type.cols;
    const rows = rotated ? type.cols : type.rows;

    // Only raycast against bricks if mouse is near the baseplate area
    let overBrick = false;
    let snapPoint = _intersectPoint;
    if (placedBricks.length > 0) {
      const brickHits = raycaster.intersectObjects(placedBricks.map(b => b.mesh), false);
      overBrick = brickHits.length > 0;
      if (overBrick) snapPoint = brickHits[0].point;
    }

    const rawGX = Math.floor((snapPoint.x + baseplateSize / 2) / UNIT);
    const rawGZ = Math.floor((snapPoint.z + baseplateSize / 2) / UNIT);

    // Check if the mouse is actually over the baseplate area
    const overBaseplate = rawGX >= 0 && rawGX < GRID_SIZE && rawGZ >= 0 && rawGZ < GRID_SIZE;

    if (overBaseplate || overBrick) {
      ghostGridX = Math.max(0, Math.min(GRID_SIZE - cols, rawGX));
      ghostGridZ = Math.max(0, Math.min(GRID_SIZE - rows, rawGZ));

      // Hide ghost if a brick is still dropping into this position
      if (isBrickDroppingAtPosition(ghostGridX, ghostGridZ, cols, rows)) {
        if (ghostBrick) ghostBrick.visible = false;
        ghostWasHidden = true;
      } else {
        // Show ghost if it was hidden
        if (ghostWasHidden && ghostBrick) {
          ghostWasHidden = false;
        }
        if (ghostBrick && !ghostBrick.visible) ghostBrick.visible = true;
        if (!ghostBrick) updateGhostBrick();
      }

      // Play tick when ghost moves to a new grid cell
      if (ghostGridX !== prevGhostGridX || ghostGridZ !== prevGhostGridZ) {
        prevGhostGridX = ghostGridX;
        prevGhostGridZ = ghostGridZ;
        playHoverTickSound();
      }

      positionGhostBrick();
    } else {
      // Hide ghost when not over baseplate or bricks
      if (ghostBrick) ghostBrick.visible = false;
    }
  } else {
    // Ray didn't hit the plane at all — hide ghost
    if (ghostBrick) ghostBrick.visible = false;
  }
}

function onMouseClick(event) {
  if (event.button !== 0) return;
  if (!placementMode) return;
  // Don't place if ghost brick is hidden (not hovering over baseplate/bricks)
  if (!ghostBrick || !ghostBrick.visible) return;
  // Don't place if clicking UI
  if (event.target.closest('#gameUI')) return;

  const type = BRICK_TYPES[selectedTypeIndex];
  const cols = rotated ? type.rows : type.cols;
  const rows = rotated ? type.cols : type.rows;
  const layer = getLayerAtPosition(ghostGridX, ghostGridZ, cols, rows);

  if (!isValidPlacement(ghostGridX, ghostGridZ, cols, rows, layer)) {
    playErrorSound();
    return;
  }

  // Place brick
  const geo = buildBrickGeometry(cols, rows);
  const mat = getBrickMaterial(BRICK_COLORS[selectedColorIndex].hex);
  const mesh = new THREE.Mesh(geo, mat);
  mesh.castShadow = true;
  mesh.receiveShadow = true;
  mesh.name = `brick_${brickCounter++}`;

  const x = -baseplateSize / 2 + ghostGridX * UNIT + (cols * UNIT) / 2;
  const z = -baseplateSize / 2 + ghostGridZ * UNIT + (rows * UNIT) / 2;
  const y = layer * BRICK_HEIGHT;
  mesh.position.set(x, y, z);

  // Drop animation
  mesh.position.y = y + 3;
  mesh.userData.targetY = y;
  mesh.userData.dropping = true;
  mesh.userData.dropSpeed = 0;

  scene.add(mesh);

  const brickData = {
    mesh, gridX: ghostGridX, gridZ: ghostGridZ, cols, rows, layer,
    colorHex: BRICK_COLORS[selectedColorIndex].hex
  };
  placedBricks.push(brickData);
  totalBricksPlaced++;

  createSnapEffect(x, y, z);
  playSnapSound();

  // Hide ghost immediately while brick is dropping to avoid overlap
  if (ghostBrick) {
    ghostBrick.visible = false;
    ghostWasHidden = true;
  }

  updateBrickCount();
  positionGhostBrick();
}

function onRightClick(event) {
  event.preventDefault();
  // Remove last brick at hovered position
  raycaster.setFromCamera(mouse, camera);
  const meshes = placedBricks.map(b => b.mesh);
  const intersects = raycaster.intersectObjects(meshes);
  if (intersects.length > 0) {
    const hitMesh = intersects[0].object;
    const idx = placedBricks.findIndex(b => b.mesh === hitMesh);
    if (idx !== -1) {
      scene.remove(hitMesh);
      hitMesh.geometry.dispose();
      placedBricks.splice(idx, 1);
      playRemoveSound();
      updateBrickCount();
      positionGhostBrick();
    }
  }
}

// ─── Snap effect (particles) ───
const snapParticles = [];

// Shared geometry and material pool for snap particles
const snapParticleGeo = new THREE.SphereGeometry(0.04, 4, 4);
const snapParticleMatPool = [];
for (let i = 0; i < 6; i++) {
  snapParticleMatPool.push(new THREE.MeshBasicMaterial({ color: 0xffffff, transparent: true, opacity: 1 }));
}

function createSnapEffect(x, y, z) {
  for (let i = 0; i < 6; i++) {
    const mat = snapParticleMatPool[i].clone();
    const p = new THREE.Mesh(snapParticleGeo, mat);
    p.position.set(x, y + BRICK_HEIGHT / 2, z);
    const angle = (i / 6) * Math.PI * 2;
    p.userData.vx = Math.cos(angle) * 2;
    p.userData.vy = 2 + Math.random() * 2;
    p.userData.vz = Math.sin(angle) * 2;
    p.userData.life = 1.0;
    p.name = `snapParticle_${i}`;
    scene.add(p);
    snapParticles.push(p);
  }
}

// ─── Keyboard ───
window.addEventListener('keydown', (e) => {
  const key = e.key.toLowerCase();
  if (key === 'r') {
    rotated = !rotated;
    updateGhostBrick();
    updateRotateBtn();
    playRotateSound();
  }
  if (key === 'p') {
    togglePlacementMode();
  }
  if (key === 'z' && e.ctrlKey && placedBricks.length > 0) {
    const last = placedBricks.pop();
    scene.remove(last.mesh);
    last.mesh.geometry.dispose();
    playRemoveSound();
    updateBrickCount();
    positionGhostBrick();
  }
  // Number keys to select brick type
  if (key >= '1' && key <= '7') {
    selectedTypeIndex = parseInt(key) - 1;
    rotated = false;
    updateGhostBrick();
    updateTypeButtons();
    playSelectSound();
  }
});

renderer.domElement.addEventListener('pointerdown', onMouseClick);
renderer.domElement.addEventListener('contextmenu', onRightClick);

function applyToAllBrickMaterials(fn) {
  for (const key in materialCache) {
    if (key.endsWith('_0.45')) continue; // skip ghost materials
    fn(materialCache[key], key);
  }
}

// Helper: get luminance tier for a material cache key (the hex color)
function getLuminanceTier(cacheKey) {
  const hex = cacheKey.split('_')[0];
  const c = new THREE.Color(hex);
  const lum = 0.2126 * c.r + 0.7152 * c.g + 0.0722 * c.b;
  if (lum > 0.7) return 'veryLight';
  if (lum > 0.45) return 'light';
  return 'normal';
}

function sliderBind(id, valId, divisor, callback) {
  const slider = document.getElementById(id);
  const valEl = document.getElementById(valId);
  slider.addEventListener('input', () => {
    const v = parseFloat(slider.value) / divisor;
    if (valEl) valEl.textContent = v.toFixed(2);
    callback(v);
  });
}

sliderBind('propRoughness', 'propRoughnessVal', 100, (v) => {
  applyToAllBrickMaterials((m, key) => {
    const tier = getLuminanceTier(key);
    // Very light bricks always get extra roughness to tame highlights
    const bump = tier === 'veryLight' ? 0.35 : tier === 'light' ? 0.1 : 0;
    m.roughness = Math.min(1, v + bump);
  });
  baseplateMat.roughness = Math.min(1, v + 0.3);
});

// Debounce timer for env rotation (pixel processing is expensive)
let envRotationTimer = null;

function applyEnvOrientation() {
  // Rotate rim light with environment (key light is now independent via shadow controls)
  const envRotationRad = envRotationDeg * Math.PI / 180;
  const envTiltRad = envTiltDeg * Math.PI / 180;
  const cosR = Math.cos(envRotationRad), sinR = Math.sin(envRotationRad);
  const cosT = Math.cos(envTiltRad), sinT = Math.sin(envTiltRad);

  let rx = rimLightInitPos.x * cosR + rimLightInitPos.z * sinR;
  let ry = rimLightInitPos.y;
  let rz = -rimLightInitPos.x * sinR + rimLightInitPos.z * cosR;
  let ry2 = ry * cosT - rz * sinT;
  let rz2 = ry * sinT + rz * cosT;
  rimLight.position.set(rx, ry2, rz2);

  // Debounced HDR pixel rotation (re-processes the raw equirect HDR)
  clearTimeout(envRotationTimer);
  envRotationTimer = setTimeout(() => {
    applyHdrWithRotation(currentRawHdr, envRotationDeg, envTiltDeg);
  }, 80);
}

// ─── Preset system ───
let presetAnimating = false;
let presetAnimBricks = []; // { mesh, targetY, layer, delay, elapsed, duration, done }
const REVEAL_LAYER_STAGGER = 0.12; // seconds between layer waves starting
const REVEAL_BRICK_DURATION = 0.35; // seconds per brick animation
const REVEAL_DROP_HEIGHT = 4.0; // how far above target bricks start

function placePresetBrick(gx, gz, cols, rows, layer, colorHex) {
  const geo = buildBrickGeometry(cols, rows);
  const mat = getBrickMaterial(colorHex);
  const mesh = new THREE.Mesh(geo, mat);
  mesh.castShadow = true;
  mesh.receiveShadow = true;
  mesh.name = `brick_${brickCounter++}`;
  const x = -baseplateSize / 2 + gx * UNIT + (cols * UNIT) / 2;
  const z = -baseplateSize / 2 + gz * UNIT + (rows * UNIT) / 2;
  const y = layer * BRICK_HEIGHT;

  // Start hidden — will be revealed by animation
  mesh.position.set(x, y + REVEAL_DROP_HEIGHT, z);
  mesh.scale.setScalar(0);
  mesh.visible = false;

  scene.add(mesh);
  const brickData = { mesh, gridX: gx, gridZ: gz, cols, rows, layer, colorHex };
  placedBricks.push(brickData);
  totalBricksPlaced++;

  // Store animation info on the mesh for the reveal pass
  mesh.userData._presetReveal = { targetY: y, layer };
}

function startPresetRevealAnimation() {
  // Gather all bricks that need reveal animation
  presetAnimBricks = [];
  for (const b of placedBricks) {
    const info = b.mesh.userData._presetReveal;
    if (!info) continue;
    // Small random offset within layer so bricks don't all move in lockstep
    const layerDelay = info.layer * REVEAL_LAYER_STAGGER;
    const jitter = Math.random() * 0.08;
    presetAnimBricks.push({
      mesh: b.mesh,
      targetY: info.targetY,
      layer: info.layer,
      delay: layerDelay + jitter,
      elapsed: 0,
      duration: REVEAL_BRICK_DURATION,
      done: false,
    });
    delete b.mesh.userData._presetReveal;
  }
  presetAnimating = presetAnimBricks.length > 0;
}

// Ease-out back (slight overshoot for a satisfying snap)
function easeOutBack(t) {
  const c1 = 1.70158;
  const c3 = c1 + 1;
  return 1 + c3 * Math.pow(t - 1, 3) + c1 * Math.pow(t - 1, 2);
}

function updatePresetRevealAnimation(delta) {
  if (!presetAnimating) return;
  let allDone = true;
  for (const anim of presetAnimBricks) {
    if (anim.done) continue;
    anim.elapsed += delta;
    if (anim.elapsed < anim.delay) {
      allDone = false;
      continue;
    }
    const localT = anim.elapsed - anim.delay;
    const progress = Math.min(localT / anim.duration, 1);
    const eased = easeOutBack(progress);

    // Make visible as soon as animation starts
    anim.mesh.visible = true;

    // Scale: 0 → 1
    anim.mesh.scale.setScalar(eased);

    // Y position: targetY + DROP_HEIGHT → targetY
    anim.mesh.position.y = anim.targetY + REVEAL_DROP_HEIGHT * (1 - eased);

    if (progress >= 1) {
      anim.mesh.position.y = anim.targetY;
      anim.mesh.scale.setScalar(1);
      anim.done = true;
    } else {
      allDone = false;
    }
  }
  if (allDone) {
    presetAnimating = false;
    presetAnimBricks = [];
  }
}

function clearAllBricks() {
  presetAnimating = false;
  presetAnimBricks = [];
  placedBricks.forEach(b => { scene.remove(b.mesh); b.mesh.geometry.dispose(); });
  placedBricks = [];
  totalBricksPlaced = 0;
  brickCounter = 0;
}



// ─── Torii Gate on Island preset (32×32) ───
function buildToriiPreset() {
  // Color palette
  const cRed = '#d42020';
  const cDkRed = '#a01818';
  const cBlack = '#1b1b1b';
  const cGrey = '#6b6e6a';
  const cDkGrey = '#4a4d4a';
  const cGreen = '#00852b';
  const cBGreen = '#58ab41';
  const cBlue = '#1e5aa8';
  const cDkBlue = '#143a6b';
  const cWhite = '#b4b2a2';
  const cYellow = '#f5c518';
  const cOrange = '#e66a2a';
  const cBrown = '#6b4226';

  const GS = 32;
  const cx = GS / 2;
  const cz = GS / 2;
  const radius = 13;

  // Occupancy tracker — prevents z-fighting / overlapping bricks
  const occupied = new Set();
  const occKey = (c, r, ly) => c + ',' + r + ',' + ly;
  const canPlace = (gx, gz, cols, rows, ly) => {
    for (let dr = 0; dr < rows; dr++)
      for (let dc = 0; dc < cols; dc++)
        if (occupied.has(occKey(gx + dc, gz + dr, ly))) return false;
    return true;
  };
  const markOccupied = (gx, gz, cols, rows, ly) => {
    for (let dr = 0; dr < rows; dr++)
      for (let dc = 0; dc < cols; dc++)
        occupied.add(occKey(gx + dc, gz + dr, ly));
  };
  const safeBrick = (gx, gz, cols, rows, ly, color) => {
    if (gx < 0 || gz < 0 || gx + cols > GS || gz + rows > GS) return;
    if (!canPlace(gx, gz, cols, rows, ly)) return;
    markOccupied(gx, gz, cols, rows, ly);
    placePresetBrick(gx, gz, cols, rows, ly, color);
  };

  const isIsland = (c, r) => {
    const dx = c + 0.5 - cx, dz = r + 0.5 - cz;
    return Math.sqrt(dx * dx + dz * dz) <= radius;
  };

  // ── Layer 0: Water base (4x2 bricks for strong structure) ──
  for (let r = 0; r < GS; r += 2) {
    for (let c = 0; c < GS; c += 4) {
      const wCol = ((c + r) % 8 < 4) ? cDkBlue : cBlue;
      safeBrick(c, r, 4, 2, 0, wCol);
    }
  }

  // ── Island surface: layer 1 (4x2 bricks, offset rows for interlocking) ──
  // Pass 1: 4x2 bricks where all cells fit
  const placedIsland = new Set();
  for (let r = 0; r < GS; r += 2) {
    const off = (r % 4 === 0) ? 0 : 2; // offset every other row for brick bond
    for (let c = off; c < GS; c += 4) {
      let allIsland = true;
      for (let dr = 0; dr < 2; dr++)
        for (let dc = 0; dc < 4; dc++)
          if (!isIsland(c + dc, r + dr)) allIsland = false;
      if (!allIsland) continue;
      const dx = c + 2 - cx, dz = r + 1 - cz;
      const dist = Math.sqrt(dx * dx + dz * dz);
      const col = dist > radius - 2.5 ? cDkGrey : cGrey;
      safeBrick(c, r, 4, 2, 1, col);
      for (let dr = 0; dr < 2; dr++)
        for (let dc = 0; dc < 4; dc++)
          placedIsland.add((c + dc) + ',' + (r + dr));
    }
  }
  // Pass 2: fill remaining island cells with 2x2
  for (let r = 0; r < GS; r += 2) {
    for (let c = 0; c < GS; c += 2) {
      if (placedIsland.has(c + ',' + r)) continue;
      if (isIsland(c, r) && isIsland(c + 1, r) && isIsland(c, r + 1) && isIsland(c + 1, r + 1)) {
        safeBrick(c, r, 2, 2, 1, cDkGrey);
        for (let dr = 0; dr < 2; dr++)
          for (let dc = 0; dc < 2; dc++)
            placedIsland.add((c + dc) + ',' + (r + dr));
      }
    }
  }
  // Pass 3: fill stragglers with 1x1
  for (let r = 0; r < GS; r++) {
    for (let c = 0; c < GS; c++) {
      if (isIsland(c, r) && !placedIsland.has(c + ',' + r)) {
        safeBrick(c, r, 1, 1, 1, cDkGrey);
      }
    }
  }

  // ── Gravel path: cols 15-16, layer 2 (4x2 bricks) ──
  for (let r = 5; r <= 26; r += 2) {
    safeBrick(14, r, 4, 2, 2, cWhite);
  }

  // ── Single Torii Gate (centered, shorter — pillars layer 3-10) ──
  const gRow = 14;
  const pL = 12;  // left pillar col
  const pR = 19;  // right pillar col

  // Stone foundation — 4x2 bricks, layers 2-3
  safeBrick(pL - 1, gRow - 1, 4, 2, 2, cDkGrey);
  safeBrick(pL - 1, gRow + 1, 4, 2, 2, cDkGrey);
  safeBrick(pR - 1, gRow - 1, 4, 2, 2, cDkGrey);
  safeBrick(pR - 1, gRow + 1, 4, 2, 2, cDkGrey);
  safeBrick(pL - 1, gRow - 1, 4, 2, 3, cGrey);
  safeBrick(pL - 1, gRow + 1, 4, 2, 3, cGrey);
  safeBrick(pR - 1, gRow - 1, 4, 2, 3, cGrey);
  safeBrick(pR - 1, gRow + 1, 4, 2, 3, cGrey);

  // Left pillar — 2x2 interlocking, layers 4-10
  for (let ly = 4; ly <= 10; ly++) {
    safeBrick(pL, gRow, 2, 2, ly, cRed);
  }
  // Right pillar — 2x2, layers 4-10
  for (let ly = 4; ly <= 10; ly++) {
    safeBrick(pR, gRow, 2, 2, ly, cRed);
  }

  // Nuki (tie beam) — 4x2 bricks across, layers 8-9
  for (let c = pL; c <= pR; c += 4) {
    const w = Math.min(4, pR + 2 - c);
    safeBrick(c, gRow, w, 2, 8, cBlack);
  }
  // fill remaining nuki gap
  safeBrick(pL, gRow, 4, 2, 9, cBlack);
  safeBrick(pL + 4, gRow, 4, 2, 9, cBlack);

  // Gakuzuka (name tablet) — center, layer 10
  safeBrick(15, gRow, 2, 2, 10, cWhite);

  // Kasagi (main top beam) — 4x2 bricks, layer 11
  for (let c = pL - 2; c <= pR + 2; c += 4) {
    safeBrick(c, gRow - 1, 4, 2, 11, cRed);
    safeBrick(c, gRow + 1, 4, 2, 11, cRed);
  }
  // Shimaki (upper beam) — layer 12
  for (let c = pL - 1; c <= pR + 1; c += 4) {
    safeBrick(c, gRow, 4, 2, 12, cDkRed);
  }
  // Top ridge — layer 13
  safeBrick(pL, gRow, 4, 2, 13, cDkRed);
  safeBrick(pL + 4, gRow, 4, 2, 13, cDkRed);

  // Wing tips
  safeBrick(pL - 3, gRow, 2, 2, 12, cDkRed);
  safeBrick(pR + 2, gRow, 2, 2, 12, cDkRed);
  safeBrick(pL - 4, gRow, 2, 2, 13, cBlack);
  safeBrick(pR + 3, gRow, 2, 2, 13, cBlack);

  // ── Shrine building (rows 22-27, 4x2 brick walls) ──
  // Platform layer 2
  for (let c = 10; c <= 20; c += 4) {
    for (let r = 22; r <= 26; r += 2) {
      safeBrick(c, r, 4, 2, 2, cDkGrey);
    }
  }
  // Walls layers 3-6 (4x2 bricks for structure)
  for (let ly = 3; ly <= 6; ly++) {
    // Back wall
    for (let c = 10; c <= 20; c += 4) safeBrick(c, 27, 4, 1, ly, cRed);
    // Side walls — 2x2 stacked, offset each layer
    const sOff = ly % 2;
    safeBrick(10, 22 + sOff, 2, 2, ly, cRed);
    safeBrick(10, 24 + sOff, 2, 2, ly, cRed);
    safeBrick(20, 22 + sOff, 2, 2, ly, cRed);
    safeBrick(20, 24 + sOff, 2, 2, ly, cRed);
    // Front wall (gap for door in middle)
    safeBrick(10, 22, 4, 1, ly, cRed);
    safeBrick(18, 22, 4, 1, ly, cRed);
  }
  // Door lintel
  safeBrick(14, 22, 4, 1, 6, cBlack);

  // Shrine roof — 4x2 bricks, layers 7-8
  for (let c = 9; c <= 21; c += 4) {
    safeBrick(c, 21, 4, 2, 7, cBlack);
    safeBrick(c, 23, 4, 2, 7, cBlack);
    safeBrick(c, 25, 4, 2, 7, cBlack);
    safeBrick(c, 27, 4, 2, 7, cBlack);
  }
  for (let c = 10; c <= 20; c += 4) {
    safeBrick(c, 22, 4, 2, 8, cDkGrey);
    safeBrick(c, 24, 4, 2, 8, cDkGrey);
    safeBrick(c, 26, 4, 2, 8, cDkGrey);
  }
  safeBrick(14, 23, 4, 2, 9, cDkRed);

  // ── Stone lanterns (stacked on island surface layer 2) ──
  function placeLantern(col, row) {
    safeBrick(col, row, 2, 2, 2, cDkGrey);
    safeBrick(col, row, 2, 2, 3, cGrey);
    safeBrick(col, row, 2, 2, 4, cYellow);
    safeBrick(col, row, 2, 2, 5, cDkGrey);
  }
  placeLantern(12, 10); placeLantern(18, 10);
  placeLantern(12, 18); placeLantern(18, 18);

  // ── Trees (rooted on island surface, 2x2 trunk bricks) ──
  function placeTree(col, row, tall) {
    if (!isIsland(col, row) || !isIsland(col + 1, row) ||
        !isIsland(col, row + 1) || !isIsland(col + 1, row + 1)) return;
    const h = tall ? 4 : 2;
    for (let ly = 2; ly < 2 + h; ly++) safeBrick(col, row, 2, 2, ly, cBrown);
    const cb = 2 + h;
    safeBrick(col - 1, row - 1, 4, 2, cb, cGreen);
    safeBrick(col - 1, row + 1, 4, 2, cb, cGreen);
    safeBrick(col, row, 2, 2, cb + 1, cBGreen);
  }
  placeTree(5, 7, true);   placeTree(5, 12, false);
  placeTree(6, 20, true);  placeTree(5, 25, false);

  placeTree(5, 4, true);   placeTree(24, 4, true);

  // ── Stepping stones ──
  safeBrick(14, 3, 4, 2, 1, cDkGrey);
  safeBrick(14, 1, 4, 2, 1, cGrey);

  // ── Bushes along path (2x2 on surface) ──
  for (let r = 7; r <= 19; r += 4) {
    safeBrick(12, r, 2, 2, 2, cBGreen);
    safeBrick(18, r, 2, 2, 2, cBGreen);
  }

  // ── Komainu (guardian statues) flanking gate ──
  safeBrick(12, 12, 2, 2, 2, cDkGrey);
  safeBrick(12, 12, 2, 2, 3, cWhite);
  safeBrick(18, 12, 2, 2, 2, cDkGrey);
  safeBrick(18, 12, 2, 2, 3, cWhite);

  // ── Offering table inside shrine ──
  safeBrick(14, 25, 4, 2, 3, cOrange);
  safeBrick(14, 25, 4, 2, 4, cDkRed);

  // ── Fence along approach (4x2 bricks) ──
  for (let c = 8; c <= 22; c += 4) {

    if (!isIsland(c, 6) || !isIsland(c + 3, 7)) continue;
    safeBrick(c, 6, 4, 2, 2, cRed);
    safeBrick(c, 6, 4, 2, 3, cRed);

  safeBrick(6, 4, 4, 2, 2, cDkGrey);
  safeBrick(22, 4, 4, 2, 2, cDkGrey);

// ─── Pyramids of Giza preset (32×32) ───
function buildPyramidsPreset() {
  const cSand = '#e0c878';
  const cDkSand = '#c8a84c';
  const cLtSand = '#f0dca0';
  const cBrown = '#6b4226';
  const cDkBrown = '#4a2e18';
  const cBlue = '#1e5aa8';
  const cDkBlue = '#143a6b';
  const cLtBlue = '#4a90d9';
  const cGreen = '#00852b';
  const cBGreen = '#58ab41';
  const cWhite = '#b4b2a2';
  const cGrey = '#6b6e6a';
  const cDkGrey = '#4a4d4a';
  const cYellow = '#f5c518';
  const cRed = '#d42020';
  const cOrange = '#e66a2a';

  const GS = 32;
  const occupied = new Set();
  const occKey = (c, r, ly) => c + ',' + r + ',' + ly;
  const canPlace = (gx, gz, cols, rows, ly) => {
    for (let dr = 0; dr < rows; dr++)
      for (let dc = 0; dc < cols; dc++)
        if (occupied.has(occKey(gx + dc, gz + dr, ly))) return false;
    return true;
  };
  const markOccupied = (gx, gz, cols, rows, ly) => {
    for (let dr = 0; dr < rows; dr++)
      for (let dc = 0; dc < cols; dc++)
        occupied.add(occKey(gx + dc, gz + dr, ly));
  };
  const safeBrick = (gx, gz, cols, rows, ly, color) => {
    if (gx < 0 || gz < 0 || gx + cols > GS || gz + rows > GS) return;
    if (!canPlace(gx, gz, cols, rows, ly)) return;
    markOccupied(gx, gz, cols, rows, ly);
    placePresetBrick(gx, gz, cols, rows, ly, color);
  };

  // ── Layer 0: Sand desert floor (4x2 bricks) ──
  for (let r = 0; r < GS; r += 2) {
    for (let c = 0; c < GS; c += 4) {
      const col = ((c + r) % 8 < 4) ? cSand : cLtSand;
      safeBrick(c, r, 4, 2, 0, col);
    }
  }

  // ── Nile River (cols 0-3, flowing top to bottom) ──
  for (let r = 0; r < GS; r += 2) {
    const wc = (r % 4 === 0) ? cDkBlue : cBlue;
    safeBrick(0, r, 4, 2, 1, wc);
  }
  // River bank vegetation
  for (let r = 0; r < GS; r += 4) {
    safeBrick(4, r, 2, 2, 1, cGreen);
  }

  // ── Great Pyramid (centered at col 16, row 10, base 14x14) ──
  const p1x = 10, p1z = 4;
  const p1Size = 14;
  for (let ly = 0; ly < 7; ly++) {
    const inset = ly * 1;
    const sz = p1Size - inset * 2;
    if (sz <= 0) break;
    const sx = p1x + inset, sz0 = p1z + inset;
    // Build this layer with 4x2 bricks where possible
    for (let r = sz0; r < sz0 + sz; r += 2) {
      for (let c = sx; c < sx + sz; c += 4) {
        const w = Math.min(4, sx + sz - c);
        const h = Math.min(2, sz0 + sz - r);
        const col = (ly >= 5) ? cYellow : ((ly + c + r) % 3 === 0 ? cDkSand : cSand);
        safeBrick(c, r, w, h, ly + 1, col);
      }
    }
  }
  // Pyramid capstone
  safeBrick(p1x + 6, p1z + 6, 2, 2, 8, cYellow);

  // ── Medium Pyramid (col 24, row 6, base 10x10) ──
  const p2x = 22, p2z = 6;
  const p2Size = 10;
  for (let ly = 0; ly < 5; ly++) {
    const inset = ly * 1;
    const sz = p2Size - inset * 2;
    if (sz <= 0) break;
    const sx = p2x + inset, sz0 = p2z + inset;
    for (let r = sz0; r < sz0 + sz; r += 2) {
      for (let c = sx; c < sx + sz; c += 4) {
        const w = Math.min(4, sx + sz - c);
        const h = Math.min(2, sz0 + sz - r);
        const col = (ly >= 4) ? cYellow : ((ly + c + r) % 3 === 0 ? cDkSand : cSand);
        safeBrick(c, r, w, h, ly + 1, col);
      }
    }
  }
  safeBrick(p2x + 4, p2z + 4, 2, 2, 6, cYellow);

  // ── Small Pyramid (col 24, row 20, base 6x6) ──
  const p3x = 25, p3z = 19;
  const p3Size = 6;
  for (let ly = 0; ly < 3; ly++) {
    const inset = ly * 1;
    const sz = p3Size - inset * 2;
    if (sz <= 0) break;
    const sx = p3x + inset, sz0 = p3z + inset;
    for (let r = sz0; r < sz0 + sz; r += 2) {
      for (let c = sx; c < sx + sz; c += 4) {
        const w = Math.min(4, sx + sz - c);
        const h = Math.min(2, sz0 + sz - r);
        safeBrick(c, r, w, h, ly + 1, ((ly + c + r) % 2 === 0 ? cSand : cDkSand));
      }
    }
  }
  safeBrick(p3x + 2, p3z + 2, 2, 2, 4, cYellow);

  // ── Sphinx (col 8-14, row 20-24) ──
  // Body (lying down)
  for (let r = 21; r < 25; r += 2) {
    safeBrick(8, r, 4, 2, 1, cDkSand);
    safeBrick(12, r, 4, 2, 1, cDkSand);
  }
  // Paws (front)
  safeBrick(7, 21, 1, 1, 1, cDkSand);
  safeBrick(7, 24, 1, 1, 1, cDkSand);
  safeBrick(6, 21, 2, 1, 1, cSand);
  safeBrick(6, 24, 2, 1, 1, cSand);
  // Body layer 2
  safeBrick(9, 22, 4, 2, 2, cSand);
  safeBrick(13, 22, 2, 2, 2, cSand);
  // Head (taller at back end)
  safeBrick(14, 22, 2, 2, 2, cDkSand);
  safeBrick(14, 22, 2, 2, 3, cSand);
  safeBrick(14, 22, 2, 2, 4, cDkSand);
  // Headdress
  safeBrick(13, 22, 1, 1, 3, cYellow);
  safeBrick(13, 23, 1, 1, 3, cYellow);

  // ── Obelisk near river ──
  safeBrick(5, 14, 2, 2, 1, cGrey);
  safeBrick(5, 14, 2, 2, 2, cGrey);
  safeBrick(5, 14, 2, 2, 3, cWhite);
  safeBrick(5, 14, 2, 2, 4, cWhite);
  safeBrick(5, 14, 2, 2, 5, cWhite);
  safeBrick(5, 14, 2, 2, 6, cYellow);

  // ── Palm trees ──
  function placePalm(col, row) {
    safeBrick(col, row, 1, 1, 1, cBrown);
    safeBrick(col, row, 1, 1, 2, cBrown);
    safeBrick(col, row, 1, 1, 3, cBrown);
    safeBrick(col, row, 1, 1, 4, cBrown);
    safeBrick(col - 1, row - 1, 2, 1, 5, cGreen);
    safeBrick(col, row - 1, 2, 1, 5, cGreen);
    safeBrick(col - 1, row + 1, 2, 1, 5, cGreen);
    safeBrick(col, row + 1, 2, 1, 5, cGreen);
    safeBrick(col, row, 1, 1, 5, cBGreen);
  }
  placePalm(5, 4);
  placePalm(5, 9);
  placePalm(5, 20);
  placePalm(5, 27);

  // ── Camel caravan (row 28-29) ──
  function placeCamel(col, row) {
    // Legs
    safeBrick(col, row, 1, 1, 1, cBrown);
    safeBrick(col + 2, row, 1, 1, 1, cBrown);
    // Body
    safeBrick(col, row, 4, 1, 2, cDkSand);
    // Hump
    safeBrick(col + 1, row, 2, 1, 3, cBrown);
    // Head
    safeBrick(col + 3, row, 1, 1, 3, cDkSand);
  }
  placeCamel(9, 28);
  placeCamel(16, 28);
  placeCamel(23, 29);

  // ── Sand dunes (organic mounds) ──
  safeBrick(18, 24, 4, 2, 1, cLtSand);
  safeBrick(20, 26, 4, 2, 1, cLtSand);
  safeBrick(19, 25, 2, 2, 2, cSand);

  safeBrick(8, 16, 4, 2, 1, cLtSand);
  safeBrick(10, 18, 4, 2, 1, cLtSand);
  safeBrick(9, 17, 2, 2, 2, cSand);

  // ── Path from river to pyramids ──
  for (let c = 4; c <= 9; c += 4) {
    safeBrick(c, 10, 4, 2, 1, cWhite);
  }
}

// ─── Family House preset (32×32) ───
// Clean suburban house with proper openings for windows and doors
function buildFamilyHousePreset() {
  const cWhite = '#f2f0e6';
  const cCream = '#e8dcc8';
  const cTan = '#d4b880';
  const cRoofRed = '#c02020';
  const cDkRoofRed = '#901818';
  const cBlue = '#1e5aa8';
  const cBrown = '#6b4226';
  const cDkBrown = '#4a2e18';
  const cGreen = '#00852b';
  const cBGreen = '#58ab41';
  const cGrey = '#8a8d8a';
  const cDkGrey = '#5a5d5a';
  const cRed = '#d42020';
  const cYellow = '#f5c518';
  const cDoor = '#c06820';
  const cWindow = '#b8dcff';
  const cPink = '#e87898';
  const cFlowerRed = '#d03030';

  const GS = 32;
  const b = (gx, gz, cols, rows, ly, color) => {
    if (gx < 0 || gz < 0 || gx + cols > GS || gz + rows > GS) return;
    placePresetBrick(gx, gz, cols, rows, ly, color);
  };

  // ═══════════════════════════════════════════
  // LAYER 0: Ground — lawn
  // ═══════════════════════════════════════════
  for (let r = 0; r < GS; r += 2)
    for (let c = 0; c < GS; c += 4)
      b(c, r, 4, 2, 0, ((c + r) % 8 < 4) ? cGreen : cBGreen);

  // ═══════════════════════════════════════════
  // House: 18 wide × 14 deep, origin at (7, 14)
  // Front wall is at row hz (14), back wall at hz+hd-1 (27)
  // Left wall col hx (7), right wall col hx+hw-1 (24)
  // ═══════════════════════════════════════════
  const hx = 7, hz = 14, hw = 18, hd = 14;

  // ── Foundation slab (layer 1) ──
  for (let r = hz; r < hz + hd; r += 2)
    for (let c = hx; c < hx + hw; c += 4) {
      const w = Math.min(4, hx + hw - c);
      b(c, r, w, 2, 1, cDkGrey);
    }

  // ═══════════════════════════════════════════
  // FIRST FLOOR WALLS (layers 2-5) with openings
  // Front wall: door at cols 15-16, windows at 9-12 and 19-22
  // Back wall: door at cols 15-16, windows at 11-12 and 19-20
  // Side walls: windows at rows 18-19 and 23-24
  // ═══════════════════════════════════════════

  // Front wall openings: door area cols 14-17 (ly 2-5), win cols 9-12 and 19-22 (ly 3-4)
  const frontDoorArea = new Set([14, 15, 16, 17]); // full door area including frame
  const frontDoorOpen = new Set([15, 16]); // actual opening
  const frontWinLCols = new Set([10, 11]); // open glass area only
  const frontWinLAll  = new Set([9, 10, 11, 12]); // full window including shutters
  const frontWinRCols = new Set([20, 21]); // open glass area only
  const frontWinRAll  = new Set([19, 20, 21, 22]);
  // Back wall openings
  const backDoorOpen = new Set([15, 16]);
  const backWinLCols = new Set([11, 12]);
  const backWinRCols = new Set([19, 20]);
  // Side wall openings
  const sideWinRows = new Set([18, 19, 23, 24]);

  // Build first floor walls column by column, skipping ALL positions that will
  // get dedicated bricks (door frame, shutters, openings) to prevent z-fighting
  for (let ly = 2; ly <= 5; ly++) {
    // --- Front wall (row hz) ---
    for (let c = hx; c < hx + hw; c++) {
      // Skip entire door area (cols 14-17) at all layers — frame/lintel placed separately
      if (frontDoorArea.has(c) && ly >= 2 && ly <= 5) continue;
      // Skip entire window areas (shutters + openings) at ly 2-4
      if ((frontWinLAll.has(c) || frontWinRAll.has(c)) && ly >= 2 && ly <= 4) continue;
      b(c, hz, 1, 1, ly, ly === 5 ? cWhite : cCream);
    }
    // --- Back wall (row hz+hd-1) ---
    for (let c = hx; c < hx + hw; c++) {
      if (backDoorOpen.has(c) && ly >= 2 && ly <= 4) continue;
      if ((backWinLCols.has(c) || backWinRCols.has(c)) && ly >= 3 && ly <= 4) continue;
      b(c, hz + hd - 1, 1, 1, ly, ly === 5 ? cWhite : cCream);
    }
    // --- Left wall (col hx) ---
    for (let r = hz + 1; r < hz + hd - 1; r++) {
      if (sideWinRows.has(r) && ly >= 3 && ly <= 4) continue;
      b(hx, r, 1, 1, ly, cCream);
    }
    // --- Right wall (col hx+hw-1) ---
    for (let r = hz + 1; r < hz + hd - 1; r++) {
      if (sideWinRows.has(r) && ly >= 3 && ly <= 4) continue;
      b(hx + hw - 1, r, 1, 1, ly, cCream);
    }
  }

  // ── Front door frame + lintel (no bricks in actual opening cols 15-16) ──
  b(14, hz, 1, 1, 2, cDkBrown);  // left frame
  b(14, hz, 1, 1, 3, cDkBrown);
  b(14, hz, 1, 1, 4, cDkBrown);
  b(17, hz, 1, 1, 2, cDkBrown);  // right frame
  b(17, hz, 1, 1, 3, cDkBrown);
  b(17, hz, 1, 1, 4, cDkBrown);
  b(14, hz, 4, 1, 5, cDkBrown);  // lintel spanning full door area
  // cols 15-16 ly 2-4 are TRUE empty openings

  // ── Front left window (cols 10-11 are OPEN, 9 and 12 are shutters) ──
  b(9,  hz, 1, 1, 3, cBlue);    // left shutter
  b(9,  hz, 1, 1, 4, cBlue);
  // cols 10-11 ly 3-4 are empty (true opening)
  b(12, hz, 1, 1, 3, cBlue);    // right shutter
  b(12, hz, 1, 1, 4, cBlue);
  b(9, hz, 4, 1, 2, cWhite);    // sill

  // ── Front right window (cols 20-21 are OPEN, 19 and 22 are shutters) ──
  b(19, hz, 1, 1, 3, cBlue);
  b(19, hz, 1, 1, 4, cBlue);
  // cols 20-21 ly 3-4 are empty (true opening)
  b(22, hz, 1, 1, 3, cBlue);
  b(22, hz, 1, 1, 4, cBlue);
  b(19, hz, 4, 1, 2, cWhite);   // sill

  // ── Side windows — all EMPTY openings (no bricks placed) ──
  // left wall and right wall side windows at rows 18-19 and 23-24 ly 3-4 stay open

  // ── Back door (cols 15-16, ly 2-4) — EMPTY opening ──
  // back door opening stays empty (no bricks)

  // ── Back windows (cols 11-12 and 19-20, ly 3-4) — EMPTY openings ──
  // back window openings stay empty (no bricks)

  // ═══════════════════════════════════════════
  // SECOND FLOOR WALLS (layers 6-9) with openings
  // Front: 3 windows at cols 9-12, 14-17, 19-22 (ly 7-8)
  // Back: 2 windows at cols 11-12, 19-20 (ly 7-8)
  // Sides: windows at rows 18-19 and 23-24 (ly 7-8)
  // ═══════════════════════════════════════════
  // Second floor window areas (shutters + openings combined)
  const f2WinAll = new Set([9,10,11,12, 14,15,16,17, 19,20,21,22]);
  const f2BackWinCols = new Set([11, 12, 19, 20]);

  for (let ly = 6; ly <= 9; ly++) {
    // Front wall — skip full window areas at ly 6-8 (sill + shutter + opening)
    for (let c = hx; c < hx + hw; c++) {
      if (f2WinAll.has(c) && ly >= 6 && ly <= 8) continue;
      b(c, hz, 1, 1, ly, ly === 9 ? cWhite : cTan);
    }
    // Back wall
    for (let c = hx; c < hx + hw; c++) {
      if (f2BackWinCols.has(c) && ly >= 7 && ly <= 8) continue;
      b(c, hz + hd - 1, 1, 1, ly, ly === 9 ? cWhite : cTan);
    }
    // Left wall
    for (let r = hz + 1; r < hz + hd - 1; r++) {
      if (sideWinRows.has(r) && ly >= 7 && ly <= 8) continue;
      b(hx, r, 1, 1, ly, cTan);
    }
    // Right wall
    for (let r = hz + 1; r < hz + hd - 1; r++) {
      if (sideWinRows.has(r) && ly >= 7 && ly <= 8) continue;
      b(hx + hw - 1, r, 1, 1, ly, cTan);
    }
  }

  // ── Second floor front windows (3 windows with shutters, inner cols are OPEN) ──
  // Window 1 (left): shutters at 9,12 — cols 10-11 are open
  b(9,  hz, 1, 1, 7, cBlue); b(9,  hz, 1, 1, 8, cBlue);
  b(12, hz, 1, 1, 7, cBlue); b(12, hz, 1, 1, 8, cBlue);
  b(9, hz, 4, 1, 6, cWhite);  // sill
  // Window 2 (center): shutters at 14,17 — cols 15-16 are open
  b(14, hz, 1, 1, 7, cBlue); b(14, hz, 1, 1, 8, cBlue);
  b(17, hz, 1, 1, 7, cBlue); b(17, hz, 1, 1, 8, cBlue);
  b(14, hz, 4, 1, 6, cWhite); // sill
  // Window 3 (right): shutters at 19,22 — cols 20-21 are open
  b(19, hz, 1, 1, 7, cBlue); b(19, hz, 1, 1, 8, cBlue);
  b(22, hz, 1, 1, 7, cBlue); b(22, hz, 1, 1, 8, cBlue);
  b(19, hz, 4, 1, 6, cWhite); // sill

  // ── Second floor side windows — EMPTY openings ──
  // side window openings at rows 18-19 and 23-24 ly 7-8 stay open

  // ── Second floor back windows — EMPTY openings ──
  // back window openings at cols 11-12 and 19-20 ly 7-8 stay open

  // ═══════════════════════════════════════════
  // ROOF — stepped gable (layers 10-14)
  // ═══════════════════════════════════════════
  const roofSteps = [
    { inset: -1, ly: 10, col: cRoofRed },
    { inset: 2,  ly: 11, col: cDkRoofRed },
    { inset: 4,  ly: 12, col: cRoofRed },
    { inset: 6,  ly: 13, col: cDkRoofRed },
    { inset: 8,  ly: 14, col: cRoofRed },
  ];
  for (const step of roofSteps) {
    const sx = hx + step.inset;
    const sz = hz + Math.max(0, step.inset);
    const ew = hx + hw - step.inset;
    const ed = hz + hd - Math.max(0, step.inset);
    if (ew <= sx || ed <= sz) continue;
    for (let r = sz; r < ed; r += 2)
      for (let c = sx; c < ew; c += 4) {
        const w = Math.min(4, ew - c);
        const h = Math.min(2, ed - r);
        if (w > 0 && h > 0) b(c, r, w, h, step.ly, step.col);
      }
  }

  // ── Chimney (right side, pokes through roof) ──
  for (let ly = 10; ly <= 15; ly++)
    b(hx + hw - 4, hz + 3, 2, 2, ly, ly >= 14 ? '#1b1b1b' : cDkGrey);

  // ═══════════════════════════════════════════
  // FRONT YARD
  // ═══════════════════════════════════════════

  // ── Stone walkway from front to door (centered, cols 14-17) ──
  // Start at row 3 (avoid fence at row 2), stop before porch (hz-2)
  for (let r = 3; r < hz - 2; r += 2) {
    const h = Math.min(2, hz - 2 - r);
    if (h > 0) b(14, r, 4, h, 1, cGrey);
  }

  // ── White picket fence along row 2 ──
  for (let c = 0; c < GS; c += 2) {
    if (c >= 13 && c <= 18) continue; // gate opening
    b(c, 2, 2, 1, 1, cWhite);
    if (c % 4 === 0) b(c, 2, 1, 1, 2, cWhite); // posts
  }
  // Side fences — stop before the house foundation to avoid z-fighting
  for (let r = 2; r < hz; r += 2) {
    b(0, r, 1, 2, 1, cWhite);
    b(31, r, 1, 2, 1, cWhite);
  }

  // ── Mailbox (placed at col 12 to avoid fence overlap) ──
  b(12, 3, 1, 1, 1, cDkGrey);
  b(12, 3, 1, 1, 2, cDkGrey);
  b(12, 3, 1, 1, 3, cRed);

  // ── Front porch (in front of door, 1 row deep to avoid flower bed overlap) ──
  for (let c = hx + 4; c < hx + 14; c += 4) {
    const w = Math.min(4, hx + 14 - c);
    b(c, hz - 2, w, 1, 1, cDkBrown);
  }
  // Porch step (row hz-1, narrower to leave space for flower beds at sides)
  b(hx + 5, hz - 1, 4, 1, 1, cDkBrown);
  b(hx + 9, hz - 1, 4, 1, 1, cDkBrown);
  // Porch pillars
  b(hx + 4, hz - 2, 1, 1, 2, cWhite);
  b(hx + 4, hz - 2, 1, 1, 3, cWhite);
  b(hx + 13, hz - 2, 1, 1, 2, cWhite);

    b(col, row - 1, 2, 2, treeTop, cGreen);
    b(col - 1, row, 2, 2, treeTop, cBGreen);
    b(col, row, 2, 2, treeTop, cBGreen);
    b(col, row, 1, 1, treeTop + 1, cGreen);

  placeTree(3, 28, false);
  placeTree(29, 28, false);

  b(hx + 1, hz - 1, 4, 1, 1, cBGreen);

    if (w > 0) b(c, hz + hd, w, 2, 1, cBrown);
  }
}

// ─── Statue of Liberty preset (32×32) ───
function buildStatueOfLibertyPreset() {
  // Color palette matching the LEGO reference image
  const cTan      = '#d4c8a0';   // main pedestal tan
  const cLtTan    = '#ddd4b8';   // lighter tan highlight
  const cDkTan    = '#b8a878';   // darker tan shadow/detail
  const cMdTan    = '#c8bc94';   // medium tan
  const cGreen    = '#1a9e3f';   // statue main bright green
  const cLtGreen  = '#2ab84f';   // lighter green highlights
  const cDkGreen  = '#148032';   // darker green shadows
  const cMdGreen  = '#1e8c38';   // medium green
  const cYellow   = '#f5c518';   // torch flame
  const cOrange   = '#e8a020';   // torch flame warm
  const cGrey     = '#9a9890';   // ground/base
  const cDkGrey   = '#6a6860';   // dark ground accent

  const GS = 32;
  const occupied = new Set();
  const occKey = (c, r, ly) => c + ',' + r + ',' + ly;
  const canPlace = (gx, gz, cols, rows, ly) => {
    for (let dr = 0; dr < rows; dr++)
      for (let dc = 0; dc < cols; dc++)
        if (occupied.has(occKey(gx + dc, gz + dr, ly))) return false;
    return true;
  };
  const markOccupied = (gx, gz, cols, rows, ly) => {
    for (let dr = 0; dr < rows; dr++)
      for (let dc = 0; dc < cols; dc++)
        occupied.add(occKey(gx + dc, gz + dr, ly));
  };
  const b = (gx, gz, cols, rows, ly, color) => {
    if (gx < 0 || gz < 0 || gx + cols > GS || gz + rows > GS) return;
    if (!canPlace(gx, gz, cols, rows, ly)) return;
    markOccupied(gx, gz, cols, rows, ly);
    placePresetBrick(gx, gz, cols, rows, ly, color);
  };

  const fillRect = (x, z, w, d, ly, c1, c2) => {
    const cc = c2 || c1;
    for (let r = z; r < z + d; r += 2) {
      const off = ((r - z) % 4 === 0) ? 0 : 2;
      for (let c = x + off; c < x + w; c += 4) {
        const cw = Math.min(4, x + w - c);
        if (cw >= 2) b(c, r, cw, Math.min(2, z + d - r), ly, ((r + c) % 4 < 2) ? c1 : cc);
      }
    }
    for (let r = z; r < z + d; r += 2)
      for (let c = x; c < x + w; c += 2)
        b(c, r, 2, Math.min(2, z + d - r), ly, c1);
    for (let r = z; r < z + d; r++)
      for (let c = x; c < x + w; c++)
        b(c, r, 1, 1, ly, cc);
  };

  // ═══════════════════════════════════════════
  // GROUND PLATE (layer 0)
  // ═══════════════════════════════════════════
  for (let r = 0; r < GS; r += 2)
    for (let c = 0; c < GS; c += 4)
      b(c, r, 4, 2, 0, ((c + r) % 8 < 4) ? cGrey : cDkGrey);

  // ═══════════════════════════════════════════
  // PEDESTAL BASE TIER 1 (layers 1-2) — widest, 20x20
  // ═══════════════════════════════════════════
  const b1x = 6, b1z = 6, b1w = 20, b1d = 20;
  for (let ly = 1; ly <= 2; ly++) fillRect(b1x, b1z, b1w, b1d, ly, cDkTan, cMdTan);
  // Decorative edge studs
  for (let c = b1x; c < b1x + b1w; c += 2) { b(c, b1z, 1, 1, 3, cDkTan); b(c, b1z + b1d - 1, 1, 1, 3, cDkTan); }
  for (let r = b1z; r < b1z + b1d; r += 2) { b(b1x, r, 1, 1, 3, cDkTan); b(b1x + b1w - 1, r, 1, 1, 3, cDkTan); }
  fillRect(b1x + 1, b1z + 1, b1w - 2, b1d - 2, 3, cTan, cLtTan);

  // ═══════════════════════════════════════════
  // PEDESTAL TIER 2 (layers 4-5) — 16x16
  // ═══════════════════════════════════════════
  const b2x = 8, b2z = 8, b2w = 16, b2d = 16;
  for (let ly = 4; ly <= 5; ly++) fillRect(b2x, b2z, b2w, b2d, ly, cMdTan, cTan);
  // Pillar details
  for (let c = b2x + 2; c < b2x + b2w - 2; c += 4) {
    b(c, b2z, 1, 1, 6, cDkTan);
    b(c, b2z + b2d - 1, 1, 1, 6, cDkTan);
  }
  fillRect(b2x + 1, b2z + 1, b2w - 2, b2d - 2, 6, cTan, cLtTan);

  // ═══════════════════════════════════════════
  // PEDESTAL TIER 3 — main column (layers 7-11) — 12x12
  // ═══════════════════════════════════════════
  const b3x = 10, b3z = 10, b3w = 12, b3d = 12;
  for (let ly = 7; ly <= 11; ly++) {
    const shade = (ly % 2 === 0) ? cTan : cLtTan;
    fillRect(b3x, b3z, b3w, b3d, ly, shade, cMdTan);
  }
  // Column/pillar details
  for (let ly = 7; ly <= 11; ly++) {
    b(b3x + 2, b3z, 1, 1, ly, cDkTan); b(b3x + 5, b3z, 1, 1, ly, cDkTan);
    b(b3x + 6, b3z, 1, 1, ly, cDkTan); b(b3x + 9, b3z, 1, 1, ly, cDkTan);
    b(b3x + 2, b3z + b3d - 1, 1, 1, ly, cDkTan); b(b3x + 5, b3z + b3d - 1, 1, 1, ly, cDkTan);
    b(b3x + 6, b3z + b3d - 1, 1, 1, ly, cDkTan); b(b3x + 9, b3z + b3d - 1, 1, 1, ly, cDkTan);
    b(b3x, b3z + 2, 1, 1, ly, cDkTan); b(b3x, b3z + 5, 1, 1, ly, cDkTan);
    b(b3x, b3z + 6, 1, 1, ly, cDkTan); b(b3x, b3z + 9, 1, 1, ly, cDkTan);
    b(b3x + b3w - 1, b3z + 2, 1, 1, ly, cDkTan); b(b3x + b3w - 1, b3z + 5, 1, 1, ly, cDkTan);
    b(b3x + b3w - 1, b3z + 6, 1, 1, ly, cDkTan); b(b3x + b3w - 1, b3z + 9, 1, 1, ly, cDkTan);
  }
  // Window/door indent on front
  for (let ly = 8; ly <= 10; ly++) {
    b(b3x + 4, b3z, 1, 1, ly, cDkTan); b(b3x + 7, b3z, 1, 1, ly, cDkTan);
  }
  // Cornice/ledge
  fillRect(b3x - 1, b3z - 1, b3w + 2, b3d + 2, 12, cDkTan, cMdTan);
  fillRect(b3x - 1, b3z - 1, b3w + 2, b3d + 2, 13, cTan, cLtTan);

  // ═══════════════════════════════════════════
  // STATUE — bright green, layers 14-35
  // ═══════════════════════════════════════════
  const sx = 12, sz = 12;

  // ── Robe base flare (layers 14-15) — 8x8
  for (let ly = 14; ly <= 15; ly++) fillRect(sx, sz, 8, 8, ly, cGreen, cDkGreen);
  for (let ly = 14; ly <= 15; ly++) {
    b(sx + 2, sz, 1, 1, ly, cDkGreen); b(sx + 5, sz, 1, 1, ly, cDkGreen);
    b(sx + 3, sz + 7, 1, 1, ly, cDkGreen); b(sx + 6, sz + 7, 1, 1, ly, cDkGreen);
  }

  // ── Lower body (layers 16-18) — 6x6
  for (let ly = 16; ly <= 18; ly++) fillRect(sx + 1, sz + 1, 6, 6, ly, cGreen, cMdGreen);
  for (let ly = 16; ly <= 18; ly++) {
    b(sx + 2, sz + 1, 1, 1, ly, cDkGreen); b(sx + 5, sz + 1, 1, 1, ly, cDkGreen);
    b(sx + 3, sz + 6, 1, 1, ly, cDkGreen);
  }

  // ── Mid body/waist (layers 19-20) — 6x6
  for (let ly = 19; ly <= 20; ly++) fillRect(sx + 1, sz + 1, 6, 6, ly, cMdGreen, cGreen);

  // ── Chest/upper body (layers 21-23) — 6x6
  for (let ly = 21; ly <= 23; ly++) fillRect(sx + 1, sz + 1, 6, 6, ly, cLtGreen, cGreen);

  // ── RIGHT ARM with tablet (layers 19-23) ──
  for (let ly = 19; ly <= 23; ly++) b(sx + 7, sz + 2, 2, 2, ly, cGreen);
  // Tablet
  for (let ly = 18; ly <= 22; ly++) {
    b(sx + 8, sz + 3, 1, 1, ly, '#8a7050');
    b(sx + 8, sz + 2, 1, 1, ly, '#7a6040');
  }

  // ── LEFT ARM raised with torch ──
  // Shoulder, layers 22-23
  for (let ly = 22; ly <= 23; ly++) b(sx - 1, sz + 2, 2, 2, ly, cGreen);
  // Upper arm going up, layers 24-29
  for (let ly = 24; ly <= 29; ly++) {
    const ac = (ly % 2 === 0) ? cGreen : cMdGreen;
    b(sx - 1, sz + 2, 2, 2, ly, ac);
  }
  // Wrist/hand, layers 30-31
  b(sx - 1, sz + 2, 2, 2, 30, cLtGreen);
  b(sx - 1, sz + 2, 2, 2, 31, cLtGreen);

  // ── TORCH (layers 32-35) ──
  b(sx - 1, sz + 3, 2, 1, 32, '#6a6a4a');
  b(sx - 2, sz + 2, 4, 2, 33, cOrange);
  b(sx - 1, sz + 2, 2, 2, 34, cYellow);
  b(sx, sz + 3, 1, 1, 35, cOrange);

  // ── Shoulders / upper chest (layers 24-25) — 6x4
  for (let ly = 24; ly <= 25; ly++) fillRect(sx + 1, sz + 2, 6, 4, ly, cGreen, cLtGreen);

  // ── Neck (layer 26) — 4x4
  fillRect(sx + 2, sz + 2, 4, 4, 26, cLtGreen, cGreen);

  // ═══════════════════════════════════════════
  // HEAD (layers 27-29) — 6x6
  // ═══════════════════════════════════════════
  for (let ly = 27; ly <= 28; ly++) fillRect(sx + 1, sz + 1, 6, 6, ly, cGreen, cLtGreen);
  fillRect(sx + 1, sz + 1, 6, 6, 29, cLtGreen, cGreen);
  // Face detail
  for (let ly = 27; ly <= 29; ly++) {
    b(sx + 2, sz + 1, 1, 1, ly, cMdGreen);
    b(sx + 5, sz + 1, 1, 1, ly, cMdGreen);
  }

  // ═══════════════════════════════════════════
  // CROWN (layers 30-33) — iconic 7-ray crown
  // ═══════════════════════════════════════════
  fillRect(sx + 1, sz + 1, 6, 6, 30, cDkGreen, cGreen);
  // Crown spikes — 7 rays (layers 31-33)
  // Front ray
  b(sx + 3, sz, 2, 1, 31, cGreen); b(sx + 3, sz - 1, 2, 1, 32, cGreen); b(sx + 3, sz - 1, 2, 1, 33, cDkGreen);
  // Back ray
  b(sx + 3, sz + 7, 2, 1, 31, cGreen); b(sx + 3, sz + 8, 2, 1, 32, cGreen); b(sx + 3, sz + 8, 2, 1, 33, cDkGreen);
  // Left ray

  b(sx + 1, sz, 1, 1, 31, cGreen); b(sx, sz - 1, 1, 1, 32, cGreen); b(sx, sz - 1, 1, 1, 33, cDkGreen);

  // Back-right diagonal
  b(sx + 6, sz + 7, 1, 1, 31, cGreen); b(sx + 7, sz + 8, 1, 1, 32, cGreen);
}

    for (let r = z; r < z + d; r++)
      for (let c = x; c < x + w; c++)
        b(c, r, 1, 1, ly, cc);
  };
  // Helper: fill a ring (border) of 1-stud width
  const fillRing = (x, z, w, d, ly, c) => {
    for (let cc = x; cc < x + w; cc++) { b(cc, z, 1, 1, ly, c); b(cc, z + d - 1, 1, 1, ly, c); }

  fillRect(14, 1, 4, 11, 1, cBlue, cBlue);
  for (let r = 1; r < 12; r++) {

    b(cc, plz + pld - 1, 1, 1, 3, cLtGrey);
  }
  for (let rr = plz; rr < plz + pld; rr++) {
    b(plx, rr, 1, 1, 3, cLtGrey);
    b(plx + plw - 1, rr, 1, 1, 3, cLtGrey);
  }

  // Front steps
  fillRect(plx + 9, plz + pld, 8, 1, 2, cCream, cLtGrey);
  fillRect(plx + 10, plz + pld, 6, 1, 1, cTan, cLtGrey);

  // ═══════════════════════════════════════════
  // FOUR MINARETS — at platform corners
  // base (layer 4) → shaft → balcony → upper → dome → finial
  // ═══════════════════════════════════════════
  const buildMinaret = (mmx, mmz) => {
    // Base plinth 3×3 layers 4-5 (sits on platform layer 3)
    fillRect(mmx, mmz, 3, 3, 4, cCream, cLtGrey);
    fillRect(mmx, mmz, 3, 3, 5, cWhite, cLtGrey);
    // Main shaft 2×2 layers 6-13
    for (let ly = 6; ly <= 13; ly++) {
      const shade = (ly % 4 === 0) ? cCream : (ly % 4 === 2) ? cMarble : cWhite;
      b(mmx, mmz, 2, 2, ly, shade);
    }
    // Decorative balcony ring at mid height (layer 9) — rests on shaft
    b(mmx - 1, mmz, 1, 2, 9, cLtGrey);
    b(mmx + 2, mmz, 1, 2, 9, cLtGrey);
    b(mmx, mmz - 1, 2, 1, 9, cLtGrey);
    b(mmx, mmz + 2, 2, 1, 9, cLtGrey);
    // Top balcony platform (layer 14) — rests on shaft layer 13
    fillRect(mmx - 1, mmz - 1, 4, 4, 14, cMarble, cLtGrey);
    // Upper shaft layers 15-16
    b(mmx, mmz, 2, 2, 15, cWhite);
    b(mmx, mmz, 2, 2, 16, cCream);
    // Tiny dome cap
    b(mmx, mmz, 2, 2, 17, cWhite);
    b(mmx, mmz, 1, 1, 18, cGold);
  };

  buildMinaret(plx + 1, plz + 1);
  buildMinaret(plx + plw - 3, plz + 1);
  buildMinaret(plx + 1, plz + pld - 3);
  buildMinaret(plx + plw - 3, plz + pld - 3);

  // ═══════════════════════════════════════════
  // MAIN BUILDING BODY (layers 4-11)
  // With actual arch openings cut into each facade
  // ═══════════════════════════════════════════
  const mx = 9, mz = 15, mw = 14, md = 14;

  // Build body with arch cutouts via skip set
  // Define arch openings for each face and layer
  // Front face (z = mz + md - 1 = 28): large central iwan + 2 side arches
  // Back face  (z = mz = 15): same
  // Left face  (x = mx = 9): same
  // Right face (x = mx + mw - 1 = 22): same
  const skipSet = new Set();
  const skipKey = (sx, sz, sly) => sx + ',' + sz + ',' + sly;

  // Front / Back face arch openings (holes in z-walls)
  // Large central arch: 4 wide, layers 4-10 (narrows at top)
  // Small side arches: 2 wide each, layers 4-7
  const frontZ = mz + md - 1;  // z=28
  const backZ = mz;             // z=15
  const centerX = mx + Math.floor(mw / 2); // center at x=16

  // Depth of opening — cut 3 studs deep from each face so you can see through
  const archDepth = 3;

  for (const fz of [frontZ, backZ]) {
    // z-direction: cut from fz inward by archDepth studs
    const zDir = (fz === frontZ) ? -1 : 1;
    const zStart = fz;

    // ── Large central iwan arch ──
    for (let ly = 4; ly <= 6; ly++)
      for (let cx = centerX - 3; cx < centerX + 3; cx++)
        for (let dz = 0; dz < archDepth; dz++)
          skipSet.add(skipKey(cx, zStart + zDir * dz, ly));
    for (let ly = 7; ly <= 8; ly++)
      for (let cx = centerX - 2; cx < centerX + 2; cx++)
        for (let dz = 0; dz < archDepth; dz++)
          skipSet.add(skipKey(cx, zStart + zDir * dz, ly));
    for (let ly = 9; ly <= 10; ly++)
      for (let cx = centerX - 1; cx < centerX + 1; cx++)
        for (let dz = 0; dz < archDepth - 1; dz++)
          skipSet.add(skipKey(cx, zStart + zDir * dz, ly));

    // ── Small left side arch ──
    const slx = mx + 2;
    for (let ly = 4; ly <= 5; ly++)
      for (let cx = slx; cx < slx + 2; cx++)
        for (let dz = 0; dz < 2; dz++)
          skipSet.add(skipKey(cx, zStart + zDir * dz, ly));
    for (let ly = 6; ly <= 7; ly++)
      for (let dz = 0; dz < 2; dz++)
        skipSet.add(skipKey(slx, zStart + zDir * dz, ly));

    // ── Small right side arch ──
    const srx = mx + mw - 4;
    for (let ly = 4; ly <= 5; ly++)
      for (let cx = srx; cx < srx + 2; cx++)
        for (let dz = 0; dz < 2; dz++)
          skipSet.add(skipKey(cx, zStart + zDir * dz, ly));
    for (let ly = 6; ly <= 7; ly++)
      for (let dz = 0; dz < 2; dz++)
        skipSet.add(skipKey(srx + 1, zStart + zDir * dz, ly));
  }

  // Left / Right face arch openings (holes in x-walls)
  const leftX = mx;              // x=9
  const rightX = mx + mw - 1;   // x=22
  const centerZ = mz + Math.floor(md / 2); // center at z=22

  for (const fx of [leftX, rightX]) {
    const xDir = (fx === leftX) ? 1 : -1;
    const xStart = fx;

    // ── Large central iwan arch ──
    for (let ly = 4; ly <= 6; ly++)
      for (let cz = centerZ - 3; cz < centerZ + 3; cz++)
        for (let dx = 0; dx < archDepth; dx++)
          skipSet.add(skipKey(xStart + xDir * dx, cz, ly));
    for (let ly = 7; ly <= 8; ly++)
      for (let cz = centerZ - 2; cz < centerZ + 2; cz++)
        for (let dx = 0; dx < archDepth; dx++)
          skipSet.add(skipKey(xStart + xDir * dx, cz, ly));
    for (let ly = 9; ly <= 10; ly++)
      for (let cz = centerZ - 1; cz < centerZ + 1; cz++)
        for (let dx = 0; dx < archDepth - 1; dx++)
          skipSet.add(skipKey(xStart + xDir * dx, cz, ly));

    // ── Small left (near) side arch ──
    const slz = mz + 2;
    for (let ly = 4; ly <= 5; ly++)
      for (let cz = slz; cz < slz + 2; cz++)
        for (let dx = 0; dx < 2; dx++)
          skipSet.add(skipKey(xStart + xDir * dx, cz, ly));
    for (let ly = 6; ly <= 7; ly++)
      for (let dx = 0; dx < 2; dx++)
        skipSet.add(skipKey(xStart + xDir * dx, slz, ly));

    // ── Small right (far) side arch ──
    const srz = mz + md - 4;
    for (let ly = 4; ly <= 5; ly++)
      for (let cz = srz; cz < srz + 2; cz++)
        for (let dx = 0; dx < 2; dx++)
          skipSet.add(skipKey(xStart + xDir * dx, cz, ly));
    for (let ly = 6; ly <= 7; ly++)
      for (let dx = 0; dx < 2; dx++)
        skipSet.add(skipKey(xStart + xDir * dx, srz + 1, ly));
  }

  // Also hollow out the interior (inner chamber) so arches look through
  // Keep 2-stud-thick walls on each side
  const wallThick = 2;
  const innerX1 = mx + wallThick, innerX2 = mx + mw - wallThick;
  const innerZ1 = mz + wallThick, innerZ2 = mz + md - wallThick;
  for (let ly = 4; ly <= 11; ly++)
    for (let rr = innerZ1; rr < innerZ2; rr++)
      for (let cc = innerX1; cc < innerX2; cc++)
        skipSet.add(skipKey(cc, rr, ly));

  // Now fill the building body, skipping arch openings + hollow interior
  for (let ly = 4; ly <= 11; ly++) {
    const shade1 = (ly % 2 === 0) ? cWhite : cMarble;
    const shade2 = (ly % 2 === 0) ? cMarble : cCream;
    for (let rr = mz; rr < mz + md; rr++) {
      for (let cc = mx; cc < mx + mw; cc++) {
        if (skipSet.has(skipKey(cc, rr, ly))) continue;
        b(cc, rr, 1, 1, ly, ((cc + rr + ly) % 3 === 0) ? shade2 : shade1);
      }
    }
  }

  // ─── Chamfered (cut) corners for octagonal shape ───
  for (let ly = 4; ly <= 11; ly++) {
    b(mx, mz, 2, 2, ly, cLtGrey);
    b(mx + mw - 2, mz, 2, 2, ly, cLtGrey);
    b(mx, mz + md - 2, 2, 2, ly, cLtGrey);
    b(mx + mw - 2, mz + md - 2, 2, 2, ly, cLtGrey);
    b(mx + 1, mz + 1, 1, 1, ly, cCream);
    b(mx + mw - 2, mz + 1, 1, 1, ly, cCream);
    b(mx + 1, mz + md - 2, 1, 1, ly, cCream);
    b(mx + mw - 2, mz + md - 2, 1, 1, ly, cCream);
  }

  // ─── Arch border outlines (grey trim around openings) ───
  // Front and back faces — grey border on the stud just outside the opening
  for (const fz of [frontZ, backZ]) {
    // Central arch border columns
    for (let ly = 4; ly <= 10; ly++) {
      const hw = (ly <= 6) ? 3 : (ly <= 8) ? 2 : 1;
      b(centerX - hw - 1, fz, 1, 1, ly, cGrey);
      b(centerX + hw, fz, 1, 1, ly, cGrey);
    }
    // Central arch top keystone
    b(centerX - 1, fz, 2, 1, 11, cGrey);
    // Side arch borders
    b(mx + 1, fz, 1, 1, 4, cGrey); b(mx + 1, fz, 1, 1, 5, cGrey);
    b(mx + 4, fz, 1, 1, 4, cGrey); b(mx + 4, fz, 1, 1, 5, cGrey);
    b(mx + mw - 5, fz, 1, 1, 4, cGrey); b(mx + mw - 5, fz, 1, 1, 5, cGrey);
    b(mx + mw - 2, fz, 1, 1, 4, cGrey); b(mx + mw - 2, fz, 1, 1, 5, cGrey);
  }
  // Left and right faces
  for (const fx of [leftX, rightX]) {
    for (let ly = 4; ly <= 10; ly++) {
      const hw = (ly <= 6) ? 3 : (ly <= 8) ? 2 : 1;
      b(fx, centerZ - hw - 1, 1, 1, ly, cGrey);
      b(fx, centerZ + hw, 1, 1, ly, cGrey);
    }
    b(fx, centerZ - 1, 1, 2, 11, cGrey);
    b(fx, mz + 1, 1, 1, 4, cGrey); b(fx, mz + 1, 1, 1, 5, cGrey);
    b(fx, mz + 4, 1, 1, 4, cGrey); b(fx, mz + 4, 1, 1, 5, cGrey);
    b(fx, mz + md - 5, 1, 1, 4, cGrey); b(fx, mz + md - 5, 1, 1, 5, cGrey);
    b(fx, mz + md - 2, 1, 1, 4, cGrey); b(fx, mz + md - 2, 1, 1, 5, cGrey);
  }

  // ─── RED/BROWN INLAY DETAILS (pietra dura) ───
  for (let ly = 6; ly <= 8; ly++) {
    b(mx + 4, mz + md - 1, 1, 1, ly, cRed);
    b(mx + mw - 5, mz + md - 1, 1, 1, ly, cRed);
    b(mx + 4, mz, 1, 1, ly, cRed);
    b(mx + mw - 5, mz, 1, 1, ly, cRed);
    b(mx, mz + 4, 1, 1, ly, cRed);
    b(mx, mz + md - 5, 1, 1, ly, cRed);
    b(mx + mw - 1, mz + 4, 1, 1, ly, cRed);
    b(mx + mw - 1, mz + md - 5, 1, 1, ly, cRed);
  }
  b(mx + 5, mz + md - 1, 1, 1, 9, cDkRed);
  b(mx + 8, mz + md - 1, 1, 1, 9, cDkRed);
  b(mx + 5, mz, 1, 1, 9, cDkRed);
  b(mx + 8, mz, 1, 1, 9, cDkRed);

  // ═══════════════════════════════════════════
  // PARAPET / CORNICE (layer 12)
  // ═══════════════════════════════════════════
  fillRect(mx, mz, mw, md, 12, cCream, cLtGrey);
  // Crenellations on layer 13
  for (let cc = mx + 2; cc < mx + mw - 2; cc += 2) {
    b(cc, mz, 1, 1, 13, cWhite);
    b(cc, mz + md - 1, 1, 1, 13, cWhite);
  }
  for (let rr = mz + 2; rr < mz + md - 2; rr += 2) {
    b(mx, rr, 1, 1, 13, cWhite);
    b(mx + mw - 1, rr, 1, 1, 13, cWhite);
  }

  // ═══════════════════════════════════════════
  // MAIN DOME (layers 13-21)
  // Onion dome — sits solidly on parapet
  // ═══════════════════════════════════════════
  // Dome drum (layer 13) — full inner fill
  fillRect(mx + 1, mz + 1, mw - 2, md - 2, 13, cWhite, cCream);

  // Dome bulge — widest (layers 14-15)
  fillRect(mx, mz, mw, md, 14, cWhite, cMarble);
  fillRect(mx, mz, mw, md, 15, cMarble, cWhite);

  // Dome curve (layer 16)
  fillRect(mx + 1, mz + 1, mw - 2, md - 2, 16, cWhite, cCream);

  // Dome narrowing (layers 17-18)
  fillRect(mx + 2, mz + 2, mw - 4, md - 4, 17, cWhite, cMarble);
  fillRect(mx + 3, mz + 3, mw - 6, md - 6, 18, cWhite, cCream);

  // Dome upper (layer 19)
  fillRect(mx + 4, mz + 4, mw - 8, md - 8, 19, cWhite, cMarble);

  // Dome tip (layer 20)
  b(mx + 5, mz + 5, 4, 4, 20, cWhite);

  // Gold finial / spire (layers 21-22)
  b(mx + 6, mz + 6, 2, 2, 21, cGold);
  b(mx + 7, mz + 7, 1, 1, 22, cGold);

  // ═══════════════════════════════════════════
  // FOUR CHATTRIS (corner domed kiosks)
  // Pillars start on parapet (layer 12) — no floating
  // ═══════════════════════════════════════════
  const buildChattri = (cx, cz) => {
    // Solid base column layers 12-14 (rests on building body)
    for (let ly = 12; ly <= 14; ly++) {
      fillRect(cx, cz, 3, 3, ly, (ly % 2 === 0) ? cWhite : cCream);
    }
    // Mini dome (layer 15) — rests on column
    fillRect(cx, cz, 3, 3, 15, cMarble, cLtGrey);
    b(cx + 1, cz + 1, 1, 1, 16, cGold);
  };
  buildChattri(mx - 1, mz - 1);
  buildChattri(mx + mw - 2, mz - 1);
  buildChattri(mx - 1, mz + md - 2);
  buildChattri(mx + mw - 2, mz + md - 2);

  // ═══════════════════════════════════════════
  // SIDE WING EXTENSIONS — flanking halls
  // Start at layer 4 (on platform)
  // ═══════════════════════════════════════════
  // Left wing
  const lwx = mx - 4, lwz = mz + 3, lww = 4, lwd = 8;
  for (let ly = 4; ly <= 7; ly++) fillRect(lwx, lwz, lww, lwd, ly, cWhite, cMarble);
  fillRect(lwx, lwz, lww, lwd, 8, cCream, cLtGrey);
  b(lwx + 1, lwz + 3, 2, 2, 9, cWhite);
  b(lwx + 1, lwz + 3, 2, 2, 10, cCream);
  b(lwx + 2, lwz + 4, 1, 1, 11, cGold);

  // Right wing
  const rwx = mx + mw, rwz = mz + 3, rww = 4, rwd = 8;
  for (let ly = 4; ly <= 7; ly++) fillRect(rwx, rwz, rww, rwd, ly, cWhite, cMarble);
  fillRect(rwx, rwz, rww, rwd, 8, cCream, cLtGrey);
  b(rwx + 1, rwz + 3, 2, 2, 9, cWhite);
  b(rwx + 1, rwz + 3, 2, 2, 10, cCream);
  b(rwx + 1, rwz + 4, 1, 1, 11, cGold);

  // ═══════════════════════════════════════════
  // FRONT ENTRANCE GATEWAY
  // ═══════════════════════════════════════════
  const gx = mx + 4, gz = mz + md, gw = 6, gd = 2;
  for (let ly = 4; ly <= 8; ly++) {
    fillRect(gx, gz, gw, gd, ly, cWhite, cMarble);
  }
  fillRect(gx, gz, gw, gd, 9, cCream, cLtGrey);
  b(gx + 2, gz, 2, 1, 10, cWhite);
  b(gx + 3, gz, 1, 1, 11, cGold);
}

// ─── Animation ───
const clock = new THREE.Clock();
const fpsEl = document.getElementById('fpsIndicator');
let frameCount = 0, fpsTime = 0;

function animate() {
  const delta = clock.getDelta();

  frameCount++;
  fpsTime += delta;
  if (fpsTime >= 0.5) {
    const fps = Math.round(frameCount / fpsTime);
    if (fpsEl) {
      fpsEl.textContent = fps + ' FPS';
    }
    frameCount = 0;
    fpsTime = 0;
  }

  controls.update();

  // Update camera readouts
  if (fpsTime === 0) {
    const cpx = document.getElementById('propCamPosX');
    const cpy = document.getElementById('propCamPosY');
    const cpz = document.getElementById('propCamPosZ');
    const ctx = document.getElementById('propCamTarX');
    const cty = document.getElementById('propCamTarY');
    const ctz = document.getElementById('propCamTarZ');
    if (cpx) cpx.textContent = camera.position.x.toFixed(2);
    if (cpy) cpy.textContent = camera.position.y.toFixed(2);
    if (cpz) cpz.textContent = camera.position.z.toFixed(2);
    if (ctx) ctx.textContent = controls.target.x.toFixed(2);
    if (cty) cty.textContent = controls.target.y.toFixed(2);
    if (ctz) ctz.textContent = controls.target.z.toFixed(2);
  }

  // Update projection uniforms for SSR
  projMatU.value.copy(camera.projectionMatrix);
  projInvMatU.value.copy(camera.projectionMatrixInverse);

  // Animate preset reveal (parallel layer-by-layer assembly)
  updatePresetRevealAnimation(delta);

  // Animate dropping bricks
  for (const b of placedBricks) {
    if (b.mesh.userData.dropping) {
      b.mesh.userData.dropSpeed += 25 * delta;
      b.mesh.position.y -= b.mesh.userData.dropSpeed * delta;
      if (b.mesh.position.y <= b.mesh.userData.targetY) {
        b.mesh.position.y = b.mesh.userData.targetY;
        b.mesh.userData.dropping = false;

        // Brick just landed — check if ghost should reappear at this position
        if (ghostBrick && !ghostBrick.visible && ghostWasHidden) {
          const type = BRICK_TYPES[selectedTypeIndex];
          const cols = rotated ? type.rows : type.cols;
          const rows = rotated ? type.cols : type.rows;
          if (!isBrickDroppingAtPosition(ghostGridX, ghostGridZ, cols, rows)) {
            ghostBrick.visible = true;
            ghostWasHidden = false;
            positionGhostBrick();
          }
        }
      }
    }
  }

  // Animate snap particles
  for (let i = snapParticles.length - 1; i >= 0; i--) {
    const p = snapParticles[i];
    p.userData.life -= delta * 2.5;
    if (p.userData.life <= 0) {
      scene.remove(p);
      p.geometry.dispose();
      p.material.dispose();
      snapParticles.splice(i, 1);
      continue;
    }
    p.position.x += p.userData.vx * delta;
    p.position.y += p.userData.vy * delta;
    p.position.z += p.userData.vz * delta;
    p.userData.vy -= 8 * delta;
    p.material.opacity = p.userData.life;
    p.scale.setScalar(p.userData.life);
  }

  // Ghost brick gentle hover
  if (ghostBrick) {
    ghostBrick.position.y += Math.sin(Date.now() * 0.004) * 0.001;

    // Dynamically update DOF focus distance to track ghost brick top surface
    if (dofEnabled) {
      const type = BRICK_TYPES[selectedTypeIndex];
      const cols = rotated ? type.rows : type.cols;
      const rows = rotated ? type.cols : type.rows;
      const layer = getLayerAtPosition(ghostGridX, ghostGridZ, cols, rows);

      // Use the actual grid-snapped position (not the hovering mesh position)
      const gx = -baseplateSize / 2 + ghostGridX * UNIT + (cols * UNIT) / 2;
      const gz = -baseplateSize / 2 + ghostGridZ * UNIT + (rows * UNIT) / 2;
      // Focus on the top of the brick (stud tops) rather than the center
      const gy = layer * BRICK_HEIGHT + BRICK_HEIGHT + STUD_HEIGHT;

      const focusPoint = new THREE.Vector3(gx, gy, gz);
      // Compute view-space Z depth (distance along camera's look direction)
      const viewPos = focusPoint.applyMatrix4(camera.matrixWorldInverse);
      const viewZ = -viewPos.z; // negate because camera looks down -Z
      const targetFocus = Math.max(camera.near, viewZ);
      focusDistU.value = targetFocus;
      // Update the slider to reflect current value
      const slider = document.getElementById('propDofFocus');
      const valEl = document.getElementById('propDofFocusVal');
      if (slider) {
        slider.value = Math.min(parseFloat(slider.max), targetFocus);
      }
      if (valEl) valEl.textContent = targetFocus.toFixed(1);
    }
  }

  postProcessing.render();
}
